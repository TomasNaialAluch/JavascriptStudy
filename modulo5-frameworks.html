// Importar módulos de Node.js
const http = require('http');
const fs = require('fs');
const path = require('path');

// Crear servidor HTTP básico
const servidor = http.createServer((req, res) => {
    console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
    
    // Configurar headers
    res.setHeader('Content-Type', 'application/json');
    res.setHeader('Access-Control-Allow-Origin', '*');
    
    // Manejar diferentes rutas
    if (req.url === '/') {
        res.writeHead(200);
        res.end(JSON.stringify({
            mensaje: '¡Hola desde Node.js!',
            timestamp: new Date().toISOString(),
            version: process.version
        }));
    } else if (req.url === '/api/usuarios') {
        res.writeHead(200);
        res.end(JSON.stringify([
            { id: 1, nombre: 'Ana', email: 'ana@email.com' },
            { id: 2, nombre: 'Carlos', email: 'carlos@email.com' },
            { id: 3, nombre: 'María', email: 'maria@email.com' }
        ]));
    } else if (req.url === '/api/estadisticas') {
        res.writeHead(200);
        res.end(JSON.stringify({
            memoria: process.memoryUsage(),
            uptime: process.uptime(),
            plataforma: process.platform,
            arquitectura: process.arch
        }));
    } else {
        res.writeHead(404);
        res.end(JSON.stringify({ error: 'Ruta no encontrada' }));
    }
});

// Iniciar servidor
const puerto = process.env.PORT || 3000;
servidor.listen(puerto, () => {
    console.log(`Servidor ejecutándose en http://localhost:${puerto}`);
    console.log('Rutas disponibles:');
    console.log('  GET / - Página principal');
    console.log('  GET /api/usuarios - Lista de usuarios');
    console.log('  GET /api/estadisticas - Estadísticas del servidor');
});

// Manejo de errores
servidor.on('error', (error) => {
    console.error('Error del servidor:', error);
});

// Cierre graceful
process.on('SIGTERM', () => {
    console.log('Cerrando servidor...');
    servidor.close(() => {
        console.log('Servidor cerrado');
        process.exit(0);
    });

// Simular Express.js (simplificado)
class Express {
    constructor() {
        this.routes = [];
        this.middleware = [];
    }
    
    use(middleware) {
        this.middleware.push(middleware);
    }
    
    get(path, handler) {
        this.routes.push({ method: 'GET', path, handler });
    }
    
    post(path, handler) {
        this.routes.push({ method: 'POST', path, handler });
    }
    
    put(path, handler) {
        this.routes.push({ method: 'PUT', path, handler });
    }
    
    delete(path, handler) {
        this.routes.push({ method: 'DELETE', path, handler });
    }
    
    listen(port, callback) {
        const server = require('http').createServer((req, res) => {
            // Aplicar middleware
            this.middleware.forEach(middleware => {
                middleware(req, res, () => {});
            });
            
            // Buscar ruta
            const route = this.routes.find(r => 
                r.method === req.method && req.url.startsWith(r.path)
            );
            
            if (route) {
                route.handler(req, res);
            } else {
                res.writeHead(404);
                res.end('Ruta no encontrada');
            }
        });
        
        server.listen(port, callback);
    }
}

// Simular base de datos en memoria
let usuarios = [
    { id: 1, nombre: 'Ana', email: 'ana@email.com', activo: true },
    { id: 2, nombre: 'Carlos', email: 'carlos@email.com', activo: true },
    { id: 3, nombre: 'María', email: 'maria@email.com', activo: false }
];

let nextId = 4;

// Crear aplicación Express
const app = new Express();

// Middleware para parsing JSON
app.use((req, res, next) => {
    let body = '';
    req.on('data', chunk => {
        body += chunk.toString();
    });
    req.on('end', () => {
        try {
            req.body = body ? JSON.parse(body) : {};
        } catch (error) {
            req.body = {};
        }
        next();
    });
});

// Middleware de logging
app.use((req, res, next) => {
    console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
    next();
});

// Rutas de la API
app.get('/api/usuarios', (req, res) => {
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(usuarios));
});

app.get('/api/usuarios/:id', (req, res) => {
    const id = parseInt(req.url.split('/').pop());
    const usuario = usuarios.find(u => u.id === id);
    
    if (usuario) {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify(usuario));
    } else {
        res.writeHead(404, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Usuario no encontrado' }));
    }
});

app.post('/api/usuarios', (req, res) => {
    const { nombre, email } = req.body;
    
    if (!nombre || !email) {
        res.writeHead(400, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Nombre y email son requeridos' }));
        return;
    }
    
    const nuevoUsuario = {
        id: nextId++,
        nombre,
        email,
        activo: true
    };
    
    usuarios.push(nuevoUsuario);
    
    res.writeHead(201, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(nuevoUsuario));
});

app.put('/api/usuarios/:id', (req, res) => {
    const id = parseInt(req.url.split('/').pop());
    const usuarioIndex = usuarios.findIndex(u => u.id === id);
    
    if (usuarioIndex === -1) {
        res.writeHead(404, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Usuario no encontrado' }));
        return;
    }
    
    const { nombre, email, activo } = req.body;
    usuarios[usuarioIndex] = {
        ...usuarios[usuarioIndex],
        ...(nombre && { nombre }),
        ...(email && { email }),
        ...(activo !== undefined && { activo })
    };
    
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(usuarios[usuarioIndex]));
});

app.delete('/api/usuarios/:id', (req, res) => {
    const id = parseInt(req.url.split('/').pop());
    const usuarioIndex = usuarios.findIndex(u => u.id === id);
    
    if (usuarioIndex === -1) {
        res.writeHead(404, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ error: 'Usuario no encontrado' }));
        return;
    }
    
    usuarios.splice(usuarioIndex, 1);
    
    res.writeHead(204);
    res.end();
});

    console.log('  GET    /api/usuarios     - Listar usuarios');
    console.log('  GET    /api/usuarios/:id - Obtener usuario');
    console.log('  POST   /api/usuarios     - Crear usuario');
    console.log('  PUT    /api/usuarios/:id - Actualizar usuario');
    console.log('  DELETE /api/usuarios/:id - Eliminar usuario');

  test:
    runs-on: ubuntu-latest
    
    strategy:
      matrix:
        node-version: [16.x, 18.x, 20.x]
    
    steps:
    - uses: actions/checkout@v3
    
    - name: Use Node.js ${{ matrix.node-version }}
      uses: actions/setup-node@v3
      with:
        node-version: ${{ matrix.node-version }}
        cache: 'npm'
    
    - name: Install dependencies
      run: npm ci
    
    - name: Run linter
      run: npm run lint
    
    - name: Run tests
      run: npm run test:coverage
    
    - name: Build project
