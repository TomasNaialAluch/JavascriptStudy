Operadores

Los operadores permiten realizar operaciones con valores. JavaScript incluye operadores aritméticos, de comparación y lógicos para manipular datos.

🎯 Puntos Clave

6 operadores aritméticos
8 operadores de comparación
3 operadores lógicos
Precedencia de operadores
Type coercion importante

Teoría

Operadores Aritméticos

+ Suma
- Resta
* Multiplicación
/ División
% Módulo (resto)
** Exponenciación

Operadores de Comparación

== Igualdad débil
=== Igualdad estricta
!= Desigualdad débil
!== Desigualdad estricta
> < >= <= Comparaciones numéricas

Operadores Lógicos

&& AND lógico
|| OR lógico
! NOT lógico

Ejemplo Práctico

Operadores Básicos 
// Operadores aritméticos
let a = 10;
let b = 3;

console.log('Suma:', a + b);        // 13
console.log('Resta:', a - b);       // 7
console.log('Multiplicación:', a * b); // 30
console.log('División:', a / b);    // 3.33...
console.log('Módulo:', a % b);      // 1
console.log('Exponenciación:', a ** b); // 1000

// Operadores de comparación
let x = 5;
let y = "5";

console.log('Igualdad débil (==):', x == y);   // true
console.log('Igualdad estricta (===):', x === y); // false
console.log('Mayor que:', x > 3);    // true
console.log('Menor que:', x < 10);   // true

// Operadores lógicos
let esAdulto = true;
let tieneLicencia = false;

console.log('AND (&&):', esAdulto && tieneLicencia);  // false
console.log('OR (||):', esAdulto || tieneLicencia);   // true
console.log('NOT (!):', !esAdulto);                   // false
Precedencia y Type Coercion 
// Precedencia de operadores
console.log('Sin paréntesis:', 2 + 3 * 4);     // 14 (multiplicación primero)
console.log('Con paréntesis:', (2 + 3) * 4);   // 20

// Type coercion (conversión automática de tipos)
console.log('String + Number:', "5" + 3);      // "53" (concatenación)
console.log('Number + String:', 5 + "3");      // "53" (concatenación)
console.log('String - Number:', "5" - 3);      // 2 (resta convierte a número)
console.log('String * Number:', "5" * 3);      // 15 (multiplicación convierte a número)

// Comparaciones con type coercion
console.log('"5" == 5:', "5" == 5);           // true (conversión)
console.log('"5" === 5:', "5" === 5);         // false (sin conversión)
console.log('null == undefined:', null == undefined); // true
console.log('null === undefined:', null === undefined); // false

// Operadores de asignación
let numero = 10;
numero += 5;  // numero = numero + 5
console.log('+= 5:', numero);  // 15

numero *= 2;  // numero = numero * 2
console.log('*= 2:', numero);  // 30

// Operadores de incremento/decremento
let contador = 5;
console.log('Pre-incremento:', ++contador);  // 6
console.log('Post-incremento:', contador++); // 6 (pero contador ahora es 7)
console.log('Valor final:', contador);       // 7

Estructuras de Control

Las estructuras de control permiten que tu código tome decisiones y repita operaciones basándose en condiciones. Son fundamentales para crear lógica en JavaScript.

🎯 Puntos Clave

3 tipos de condicionales
4 tipos de bucles
Operador ternario
Break y continue
Iteración de arrays y objetos

Teoría

Condicionales

if/else Ejecuta código basado en condición
switch Evalúa múltiples casos con case
? : Operador ternario (una línea)

Bucles (Loops)

for Bucle con contador (índices)
while Mientras se cumple condición
for...of Iterar sobre VALORES de arrays
for...in Iterar sobre PROPIEDADES de objetos

Control de Flujo

break Salir del bucle
continue Saltar iteración
return Salir de función

Ejemplo Práctico

Condicionales y Bucles Básicos 
// Condicionales
let temperatura = 25;

if (temperatura > 30) {
    console.log("Hace mucho calor");
} else if (temperatura > 20) {
    console.log("Temperatura agradable");
} else if (temperatura > 10) {
    console.log("Hace fresco");
} else {
    console.log("Hace frío");
}

// Operador ternario
let mensaje = temperatura > 25 ? "Calor" : "Fresco";
console.log("Mensaje:", mensaje);

// Switch
let dia = "lunes";

switch (dia) {
    case "lunes":
        console.log("Inicio de semana");
        break;
    case "viernes":
        console.log("¡Fin de semana!");
        break;
    case "sábado":
    case "domingo":
        console.log("Día de descanso");
        break;
    default:
        console.log("Día laboral");
}

// Bucles básicos
console.log("--- For Loop ---");
for (let i = 0; i < 5; i++) {
    console.log(`Iteración ${i}`);
}

console.log("--- While Loop ---");
let contador = 0;
while (contador < 3) {
    console.log(`Contador: ${contador}`);
    contador++;
}
Iteración de Arrays y Objetos 
// Arrays
let frutas = ["manzana", "banana", "naranja", "uva"];

console.log("--- For...of (valores) ---");
for (let fruta of frutas) {
    console.log(fruta);
}

console.log("--- For tradicional con índices ---");
for (let i = 0; i < frutas.length; i++) {
    console.log(`${i}: ${frutas[i]}`);
}

// Objetos
let persona = { 
    nombre: "Juan", 
    edad: 30, 
    ciudad: "Madrid",
    profesion: "Desarrollador"
};

console.log("--- For...in (propiedades) ---");
for (let propiedad in persona) {
    console.log(`${propiedad}: ${persona[propiedad]}`);
}

// Break y Continue
console.log("--- Break y Continue ---");
for (let i = 0; i < 10; i++) {
    if (i === 3) {
        console.log("Saltando 3...");
        continue; // Salta esta iteración
    }
    if (i === 7) {
        console.log("Llegamos a 7, terminando...");
        break; // Sale del bucle
    }
    console.log(`Número: ${i}`);
}

// Bucle anidado
console.log("--- Bucle Anidado ---");
for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 2; j++) {
        console.log(`i=${i}, j=${j}`);
    }
}

  • Break y continue controlan el flujo de los bucles

Funciones

Las funciones son bloques de código reutilizables que realizan una tarea específica. Son fundamentales en JavaScript para organizar y reutilizar código.

3 tipos de funciones

Return y scope
Arrow functions
Closures importantes

Teoría

Tipos de Funciones

function Función declarada (tradicional)
let func = function() Función expresada (asignada)
() => {} Arrow function (ES6+)

Conceptos Clave

parámetros Valores que recibe la función
return Valor que devuelve la función
scope Alcance de las variables
hoisting Elevación de declaraciones

Ejemplo Práctico

Declaración Básica de Funciones 
// Función declarada - se puede usar antes de declararla
function saludar(nombre) {
    return `Hola, ${nombre}!`;
}

// Función expresada - se debe declarar antes de usar
let sumar = function(a, b) {
    return a + b;
};

// Arrow function (ES6) - más corta y moderna
let multiplicar = (a, b) => a * b;

// Uso de las funciones
console.log(saludar("María"));           // "Hola, María!"
console.log(sumar(5, 3));                // 8
console.log(multiplicar(4, 6));          // 24

// Función con parámetros por defecto
function crearUsuario(nombre, edad = 18, activo = true) {
    return {
        nombre,
        edad,
        activo
    };
}

console.log(crearUsuario("Juan", 25));   // { nombre: "Juan", edad: 25, activo: true }
console.log(crearUsuario("Ana"));        // { nombre: "Ana", edad: 18, activo: true }
Scope, Hoisting y Closures 
// HOISTING - Las funciones declaradas se "elevan"
console.log(saludarHoisted("Juan")); // ✅ Funciona - "Hola Juan!"

function saludarHoisted(nombre) {
    return `Hola ${nombre}!`;
}

// ❌ Esto NO funciona con funciones expresadas
// console.log(saludarExpressed("Ana")); // ReferenceError

let saludarExpressed = function(nombre) {
    return `Hola ${nombre}!`;
};

// SCOPE - Alcance de variables
let variableGlobal = "Soy global";

function ejemploScope() {
    let variableLocal = "Soy local";
    var variableVar = "Soy var";
    
    console.log("Dentro de la función:");
    console.log("Global:", variableGlobal);  // ✅ Accesible
    console.log("Local:", variableLocal);    // ✅ Accesible
    console.log("Var:", variableVar);        // ✅ Accesible
}

ejemploScope();

// ❌ Esto NO funciona - variableLocal no existe fuera
// console.log(variableLocal); // ReferenceError

// CLOSURE - Función que "recuerda" su entorno
function crearContador() {
    let contador = 0;
    
    return function() {
        contador++;
        return contador;
    };
}

let miContador = crearContador();
console.log(miContador()); // 1
console.log(miContador()); // 2
console.log(miContador()); // 3

// Cada contador es independiente
let otroContador = crearContador();
console.log(otroContador()); // 1 (no 4)

Arrays y Objetos

Los arrays y objetos son estructuras de datos fundamentales en JavaScript para organizar y manipular información. Los arrays almacenan listas de elementos, mientras que los objetos almacenan pares clave-valor.

🎯 Puntos Clave

Métodos de arrays
Acceso a propiedades
Destructuring
Métodos de objetos
Iteración y búsqueda

Teoría

Arrays

push(), pop() Agregar/quitar al final
map(), filter() Transformar elementos
find(), includes() Buscar elementos

Objetos

objeto.propiedad Acceso a propiedades
objeto.metodo() Métodos de objetos
{prop1, prop2} Destructuring

Ejemplo Práctico

Arrays y Objetos Básicos 
// ARRAYS - Listas de elementos
let frutas = ["manzana", "banana", "naranja"];

// Métodos básicos de arrays
frutas.push("uva");        // Agregar al final
console.log("Después de push:", frutas);

let ultima = frutas.pop(); // Quitar del final
console.log("Elemento quitado:", ultima);
console.log("Array después de pop:", frutas);

// Acceso a elementos
console.log("Primera fruta:", frutas[0]);     // "manzana"
console.log("Cantidad:", frutas.length);      // 2

// OBJETOS - Pares clave-valor
let persona = {
    nombre: "Juan",
    edad: 25,
    ciudad: "Madrid"
};

// Acceso a propiedades
console.log("Nombre:", persona.nombre);       // "Juan"
console.log("Edad:", persona["edad"]);        // 25

// Modificar propiedades
persona.edad = 26;
persona.profesion = "Desarrollador";

console.log("Persona actualizada:", persona);

// Método en objeto
let calculadora = {
    sumar: function(a, b) {
        return a + b;
    },
    multiplicar: function(a, b) {
        return a * b;
    }
};

console.log("Suma:", calculadora.sumar(5, 3));        // 8
console.log("Multiplicación:", calculadora.multiplicar(4, 6)); // 24
Métodos Avanzados y Destructuring 
// MÉTODOS DE TRANSFORMACIÓN DE ARRAYS
let numeros = [1, 2, 3, 4, 5];

// map() - Transformar cada elemento
let dobles = numeros.map(num => num * 2);
console.log("Dobles:", dobles); // [2, 4, 6, 8, 10]

// filter() - Filtrar elementos
let pares = numeros.filter(num => num % 2 === 0);
console.log("Pares:", pares); // [2, 4]

// reduce() - Reducir a un solo valor
let suma = numeros.reduce((acumulador, num) => acumulador + num, 0);
console.log("Suma total:", suma); // 15

// BÚSQUEDA EN ARRAYS
let frutas = ["manzana", "banana", "naranja", "uva"];

console.log("Índice de banana:", frutas.indexOf("banana")); // 1
console.log("¿Tiene uva?:", frutas.includes("uva"));        // true
console.log("Fruta larga:", frutas.find(fruta => fruta.length > 5)); // "naranja"

// DESTRUCTURING - Extraer propiedades
let usuario = {
    nombre: "Ana",
    edad: 30,
    ciudad: "Barcelona",
    profesion: "Diseñadora"
};

// Destructuring de objeto
let { nombre, edad, ciudad } = usuario;
console.log("Nombre:", nombre);   // "Ana"
console.log("Edad:", edad);       // 30
console.log("Ciudad:", ciudad);   // "Barcelona"

// Destructuring de array
let colores = ["rojo", "verde", "azul"];
let [primero, segundo] = colores;
console.log("Primer color:", primero);  // "rojo"
console.log("Segundo color:", segundo); // "verde"

// MÉTODOS DE OBJETO
let producto = {
    nombre: "Laptop",
    precio: 999,
    stock: 5,
    calcularTotal: function(cantidad) {
        return this.precio * cantidad;
    }
};

console.log("Total por 2:", producto.calcularTotal(2)); // 1998

// Object.keys() - Obtener todas las claves
console.log("Propiedades:", Object.keys(producto)); // ["nombre", "precio", "stock", "calcularTotal"]

// Object.values() - Obtener todos los valores
console.log("Valores:", Object.values(producto)); // ["Laptop", 999, 5, function...]

Ejercicio 2: Gestor de Lista de Compras

Principiante

Crea un sistema simple para gestionar una lista de compras usando arrays y objetos.

Requisitos:

  • Crear array para productos
  • Función para agregar productos
  • Función para mostrar lista
  • Función para calcular total
  • Usar objetos para cada producto
// Solución del Ejercicio 2: Gestor de Lista de Compras

// Array para almacenar productos
let listaCompras = [];

// Función para agregar producto
function agregarProducto(nombre, precio, cantidad = 1) {
    let producto = {
        nombre: nombre,
        precio: precio,
        cantidad: cantidad,
        total: precio * cantidad
    };
    
    listaCompras.push(producto);
    console.log(`✅ ${nombre} agregado a la lista`);
}

// Función para mostrar la lista
function mostrarLista() {
    console.log("🛒 Lista de Compras:");
    console.log("==================");
    
    for (let i = 0; i < listaCompras.length; i++) {
        let producto = listaCompras[i];
        console.log(`${i + 1}. ${producto.nombre} - $${producto.precio} x ${producto.cantidad} = $${producto.total}`);
    }
}

// Función para calcular total
function calcularTotal() {
    let total = 0;
    
    for (let producto of listaCompras) {
        total += producto.total;
    }
    
    return total;
}

// Probar el sistema
agregarProducto("Leche", 2.50, 2);
agregarProducto("Pan", 1.20, 1);
agregarProducto("Huevos", 3.00, 1);

mostrarLista();
console.log("💰 Total a pagar: $" + calcularTotal());

Ejercicio 4: Contador de Palabras

Principiante

Crea funciones para analizar texto usando métodos de string y arrays.

Requisitos:

  • Función para contar palabras
  • Función para contar caracteres
  • Función para encontrar palabra más larga
  • Usar split() y length
// Solución del Ejercicio 4: Contador de Palabras

function contarPalabras(texto) {
    // Dividir el texto en palabras (separadas por espacios)
    let palabras = texto.split(' ');
    
    // Filtrar palabras vacías (por si hay espacios dobles)
    let palabrasReales = palabras.filter(palabra => palabra.length > 0);
    
    return palabrasReales.length;
}

function contarCaracteres(texto) {
    return texto.length;
}

function palabraMasLarga(texto) {
    let palabras = texto.split(' ');
    let palabraLarga = '';
    
    for (let palabra of palabras) {
        if (palabra.length > palabraLarga.length) {
            palabraLarga = palabra;
        }
    }
    
    return palabraLarga;
}

function analizarTexto(texto) {
    console.log("📝 Análisis del texto:");
    console.log("=====================");
    console.log(`Texto: "${texto}"`);
    console.log(`Caracteres: ${contarCaracteres(texto)}`);
    console.log(`Palabras: ${contarPalabras(texto)}`);
    console.log(`Palabra más larga: "${palabraMasLarga(texto)}"`);
}

// Probar con diferentes textos
let textos = [
    "Hola mundo",
    "JavaScript es genial para programar",
    "Aprender programación es divertido y útil"
];

for (let texto of textos) {
    analizarTexto(texto);
    console.log(""); // Línea en blanco
}

// Ejemplo práctico
let miTexto = "Me gusta mucho programar en JavaScript";
analizarTexto(miTexto);

if/else - Condicionales

¿Qué es if/else?

Es la estructura de control más básica que permite que tu código tome decisiones. Evalúa una condición y ejecuta diferentes bloques de código según el resultado.

Sintaxis:

if (condición) {
    // Código si la condición es verdadera
} else if (otra_condición) {
    // Código si la segunda condición es verdadera
} else {
    // Código si ninguna condición es verdadera
}

Ejemplo Práctico:

let edad = 18;

if (edad >= 18) {
    console.log("Eres mayor de edad");
} else if (edad >= 13) {
    console.log("Eres adolescente");
} else {
    console.log("Eres menor de edad");
}

Puntos Clave:

  • La condición debe evaluar a true o false
  • Puedes tener múltiples else if
  • El else es opcional
  • Usa {} para bloques de múltiples líneas

switch/case - Múltiples Casos

¿Qué es switch y case?

switch es una estructura que evalúa una expresión y ejecuta diferentes bloques de código según el valor. case define cada posible valor y qué hacer en cada caso.

Sintaxis:

switch (expresión) {
    case valor1:
        // Código para valor1
        break;
    case valor2:
        // Código para valor2
        break;
    default:
        // Código si no coincide con ningún caso
        break;
}

¿Qué hace cada parte?

  • switch (expresión): Evalúa la expresión una sola vez
  • case valor: Compara el valor con la expresión
  • break: Sale del switch (¡MUY IMPORTANTE!)
  • default: Se ejecuta si ningún case coincide

Ejemplo Práctico:

let dia = "lunes";

switch (dia) {
    case "lunes":
        console.log("Inicio de semana");
        break;
    case "viernes":
        console.log("¡Fin de semana!");
        break;
    case "sábado":
    case "domingo":
        console.log("Día de descanso");
        break;
    default:
        console.log("Día laboral");
}

⚠️ ¡Cuidado con el break!

Sin break, el código "cae" al siguiente case (fall-through). Esto puede ser útil o causar errores.

for - Bucle con Contador

¿Qué es el bucle for?

Es un bucle que se repite un número específico de veces usando un contador. Es perfecto cuando sabes exactamente cuántas veces quieres que se ejecute el código.

Sintaxis:

for (inicialización; condición; actualización) {
    // Código que se repite
}

¿Qué hace cada parte?

  • inicialización: Se ejecuta UNA VEZ al inicio (ej: let i = 0)
  • condición: Se evalúa ANTES de cada iteración (ej: i < 5)
  • actualización: Se ejecuta DESPUÉS de cada iteración (ej: i++)

Ejemplo Paso a Paso:

for (let i = 0; i < 3; i++) {
    console.log(`Iteración ${i}`);
}

// Resultado:
// Iteración 0
// Iteración 1
// Iteración 2

Proceso paso a paso:

  1. Inicio: i = 0
  2. ¿i < 3? Sí (0 < 3) → Ejecuta código → i++ → i = 1
  3. ¿i < 3? Sí (1 < 3) → Ejecuta código → i++ → i = 2
  4. ¿i < 3? Sí (2 < 3) → Ejecuta código → i++ → i = 3
  5. ¿i < 3? No (3 < 3) → Termina el bucle

Usos Comunes:

  • Iterar arrays con índices: for (let i = 0; i < array.length; i++)
  • Repetir código N veces
  • Generar secuencias numéricas

for...of - Iterar VALORES

¿Qué es for...of?

Es un bucle que itera sobre los VALORES de elementos iterables (arrays, strings, etc.). Es más simple que el for tradicional cuando solo necesitas los valores, no los índices.

Sintaxis:

for (let elemento of iterable) {
    // Código que usa el elemento
}

Ejemplo con Array:

let frutas = ["manzana", "banana", "naranja"];

for (let fruta of frutas) {
    console.log(fruta);
}

// Resultado:
// manzana
// banana
// naranja

Comparación: for vs for...of

for tradicional (con índices):

for (let i = 0; i < frutas.length; i++) {
    console.log(`${i}: ${frutas[i]}`);
}
// Resultado:
// 0: manzana
// 1: banana
// 2: naranja

for...of (solo valores):

for (let fruta of frutas) {
    console.log(fruta);
}
// Resultado:
// manzana
// banana
// naranja

¿Cuándo usar for...of?

  • ✅ Cuando solo necesitas los valores
  • ✅ Código más limpio y legible
  • ✅ Menos propenso a errores
  • ❌ NO cuando necesitas el índice
  • ❌ NO para modificar el array original

for...in - Iterar PROPIEDADES

¿Qué es for...in?

Es un bucle que itera sobre las PROPIEDADES (claves) de un objeto. Te da acceso a los nombres de las propiedades, no a sus valores.

Sintaxis:

for (let propiedad in objeto) {
    // Código que usa la propiedad
}

Ejemplo con Objeto:

let persona = {
    nombre: "Juan",
    edad: 25,
    ciudad: "Madrid"
};

for (let propiedad in persona) {
    console.log(`${propiedad}: ${persona[propiedad]}`);
}

// Resultado:
// nombre: Juan
// edad: 25
// ciudad: Madrid

¿Cómo acceder a los valores?

Usa la notación de corchetes: objeto[propiedad]

for (let prop in persona) {
    console.log(prop);                    // nombre, edad, ciudad
    console.log(persona[prop]);          // Juan, 25, Madrid
    console.log(persona.prop);           // undefined (¡ERROR!)
}

Diferencia clave: for...of vs for...in

for...of (valores):

let array = [10, 20, 30];
for (let valor of array) {
    console.log(valor);  // 10, 20, 30
}

for...in (propiedades):

let array = [10, 20, 30];
for (let indice in array) {
    console.log(indice);  // 0, 1, 2
    console.log(array[indice]); // 10, 20, 30
}

⚠️ Advertencias importantes:

  • for...in también itera propiedades heredadas
  • El orden no está garantizado en objetos
  • Para arrays, usa for...of o for tradicional
×

while - Mientras se cumple condición

¿Qué es el bucle while?

Es un bucle que se repite mientras una condición sea verdadera. No sabes cuántas veces se ejecutará, solo sabes cuándo debe parar.

Sintaxis:

while (condición) {
    // Código que se repite
    // ¡IMPORTANTE! Modificar la condición aquí
}

Ejemplo Básico:

let contador = 0;

while (contador < 3) {
    console.log(`Contador: ${contador}`);
    contador++;  // ¡IMPORTANTE! Sin esto sería un bucle infinito
}

// Resultado:
// Contador: 0
// Contador: 1
// Contador: 2

Proceso paso a paso:

  1. ¿contador < 3? Sí (0 < 3) → Ejecuta → contador = 1
  2. ¿contador < 3? Sí (1 < 3) → Ejecuta → contador = 2
  3. ¿contador < 3? Sí (2 < 3) → Ejecuta → contador = 3
  4. ¿contador < 3? No (3 < 3) → Termina

⚠️ ¡Cuidado con los bucles infinitos!

// ❌ MAL - Bucle infinito
let i = 0;
while (i < 5) {
    console.log("Hola");
    // ¡Falta i++!
}

// ✅ BIEN - Condición que cambia
let i = 0;
while (i < 5) {
    console.log("Hola");
    i++;  // La condición cambia
}

¿Cuándo usar while?

  • ✅ Cuando no sabes cuántas iteraciones necesitas
  • ✅ Validación de entrada del usuario
  • ✅ Procesar datos hasta encontrar un valor específico
  • ❌ NO cuando sabes exactamente cuántas veces iterar (usa for)

Operador Ternario (? :)

¿Qué es el operador ternario?

Es una forma abreviada de escribir un if/else simple en una sola línea. Es perfecto para asignaciones condicionales.

Sintaxis:

condición ? valor_si_verdadero : valor_si_falso

Comparación: if/else vs Ternario

if/else tradicional:

let edad = 18;
let mensaje;

if (edad >= 18) {
    mensaje = "Mayor de edad";
} else {
    mensaje = "Menor de edad";
}

Operador ternario:

let edad = 18;
let mensaje = edad >= 18 ? "Mayor de edad" : "Menor de edad";

Ejemplos Prácticos:

// Asignación condicional
let precio = 100;
let descuento = precio > 50 ? 10 : 0;

// En template strings
let nombre = "Juan";
let saludo = `Hola ${nombre}, ${nombre.length > 3 ? "bienvenido" : "hola"}`;

// Retorno de función
function esPar(numero) {
    return numero % 2 === 0 ? "par" : "impar";
}

⚠️ No abuses del ternario:

// ❌ MAL - Muy difícil de leer
let resultado = a > b ? c > d ? "A" : "B" : c > d ? "C" : "D";

// ✅ BIEN - Usa if/else para casos complejos
let resultado;
if (a > b) {
    resultado = c > d ? "A" : "B";
} else {
    resultado = c > d ? "C" : "D";
}

¿Cuándo usar ternario?

  • ✅ Asignaciones simples (una condición)
  • ✅ Valores por defecto
  • ✅ Template strings
  • ❌ NO para lógica compleja
  • ❌ NO para múltiples condiciones

Función Declarada

¿Qué es una función declarada?

Es la forma tradicional de crear funciones en JavaScript. Se declara con la palabra clave function seguida del nombre de la función.

Sintaxis:

function nombreFuncion(parametros) {
    // código a ejecutar
    return valor;
}

Características:

  • Hoisting: Se puede usar antes de declararla
  • Nombre obligatorio: Debe tener un nombre
  • Scope global: Se crea en el scope donde se declara

Ejemplo:

// Se puede usar antes de declararla
console.log(saludar("Juan")); // ✅ Funciona

function saludar(nombre) {
    return `Hola, ${nombre}!`;
}

Función Expresada

¿Qué es una función expresada?

Es una función que se asigna a una variable. Se crea usando una expresión de función anónima (sin nombre propio).

Sintaxis:

let nombreVariable = function(parametros) {
    // código a ejecutar
    return valor;
};

¿Cuándo se usa?

  • Funciones condicionales: Cuando quieres crear diferentes funciones según una condición
  • Callbacks: Para pasar funciones como parámetros a otras funciones
  • Funciones dinámicas: Cuando el nombre de la función puede cambiar
  • Módulos: Para exportar funciones específicas

Ejemplos prácticos:

// 1. Función condicional
let operacion;
if (usuario.esAdmin) {
    operacion = function() {
        return "Acceso completo";
    };
} else {
    operacion = function() {
        return "Acceso limitado";
    };
}

// 2. Callback en setTimeout
setTimeout(function() {
    console.log("Se ejecutó después de 2 segundos");
}, 2000);

// 3. Función como parámetro
function procesarDatos(datos, callback) {
    let resultado = datos * 2;
    callback(resultado);
}

let miCallback = function(resultado) {
    console.log("El resultado es:", resultado);
};

procesarDatos(5, miCallback); // "El resultado es: 10"

Características:

  • Sin hoisting: No se puede usar antes de declararla
  • Asignada a variable: Se almacena en una variable
  • Anónima: La función no tiene nombre propio
  • Flexible: Se puede reasignar o pasar como parámetro

Comparación con función declarada:

// ❌ Esto NO funciona - función expresada
console.log(saludarExpresada("Juan")); // ReferenceError

let saludarExpresada = function(nombre) {
    return `Hola, ${nombre}!`;
};

// ✅ Esto SÍ funciona - función declarada
console.log(saludarDeclarada("Juan")); // "Hola, Juan!"

function saludarDeclarada(nombre) {
    return `Hola, ${nombre}!`;
}

Arrow Function (ES6)

¿Qué es ES6?

ES6 (ECMAScript 2015) es una versión importante de JavaScript que se lanzó en 2015. Introdujo muchas características modernas que hicieron JavaScript más potente y fácil de usar. ES6 incluye:

  • Arrow functions: Sintaxis más corta para funciones
  • let y const: Nuevas formas de declarar variables
  • Template literals: Strings con interpolación (backticks)
  • Destructuring: Extraer valores de objetos y arrays
  • Classes: Programación orientada a objetos
  • Modules: Importar y exportar código

¿Qué es una arrow function?

Es una forma más corta y moderna de escribir funciones, introducida en ES6. Usa la sintaxis => (flecha) y es muy popular en JavaScript moderno.

Sintaxis:

// Con un parámetro
let nombre = parametro => expresion;

// Con múltiples parámetros
let nombre = (param1, param2) => expresion;

// Con bloque de código
let nombre = (param1, param2) => {
    // código
    return valor;
};

¿Cuándo usar arrow functions?

  • Callbacks cortos: En map(), filter(), forEach()
  • Funciones simples: Cuando la lógica es básica
  • Event handlers: Para manejar eventos del DOM
  • Promesas: En .then() y .catch()

Ejemplos prácticos:

// 1. Con arrays (muy común)
let numeros = [1, 2, 3, 4, 5];

// Función tradicional
let dobles = numeros.map(function(num) {
    return num * 2;
});

// Arrow function (más corta)
let dobles = numeros.map(num => num * 2);

// 2. Con eventos
document.addEventListener('click', () => {
    console.log('Se hizo clic en la página');
});

// 3. Con promesas
fetch('/api/datos')
    .then(response => response.json())
    .then(data => console.log(data));

Características importantes:

  • Más corta: Sintaxis más concisa y legible
  • Sin this propio: Hereda el this del contexto padre
  • Return implícito: Si es una expresión, no necesita return
  • No se puede usar como constructor: No se puede usar con 'new'

Explicación Detallada de Características:

1. Sintaxis más concisa:

Las arrow functions reducen significativamente la cantidad de código necesario:

// Función tradicional - 3 líneas
let numeros = [1, 2, 3, 4, 5];
let dobles = numeros.map(function(numero) {
    return numero * 2;
});

// Arrow function - 1 línea
let dobles = numeros.map(numero => numero * 2);

// Ahorro: 66% menos código

2. Sin this propio (muy importante):

Las arrow functions NO crean su propio contexto de this, heredan el del contexto padre:

let persona = {
    nombre: "Ana",
    edad: 25,
    
    // ❌ PROBLEMA con función tradicional
    saludarTradicional: function() {
        setTimeout(function() {
            console.log(`Hola, soy ${this.nombre}`); // this = undefined
        }, 1000);
    },
    
    // ✅ SOLUCIÓN con arrow function
    saludarArrow: function() {
        setTimeout(() => {
            console.log(`Hola, soy ${this.nombre}`); // this = persona
        }, 1000);
    }
};

persona.saludarTradicional(); // "Hola, soy undefined"
persona.saludarArrow();       // "Hola, soy Ana"

3. Return implícito (cuando NO usar llaves):

Si la función es una sola expresión, NO necesitas return ni llaves:

// ✅ CON return implícito (sin llaves)
let sumar = (a, b) => a + b;
let esPar = num => num % 2 === 0;
let obtenerNombre = usuario => usuario.nombre;

// ❌ INCORRECTO - no necesitas return aquí
let sumar = (a, b) => { return a + b; };

// ✅ CON return explícito (con llaves)
let procesarDatos = (datos) => {
    let resultado = datos * 2;
    console.log("Procesando...");
    return resultado; // Necesario porque hay más código
};

// ✅ CON return implícito (una sola expresión)
let procesarDatos = (datos) => datos * 2;

Casos de uso reales:

// 1. Con arrays (muy común)
let precios = [10, 20, 30, 40];
let preciosConIva = precios.map(precio => precio * 1.21);
let preciosAltos = precios.filter(precio => precio > 25);

// 2. Con eventos del DOM
document.getElementById('boton').addEventListener('click', () => {
    console.log('Botón presionado');
});

// 3. Con promesas
fetch('/api/datos')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

Errores comunes a evitar:

// ❌ ERROR: No puedes usar arrow function como constructor
let Persona = (nombre) => {
    this.nombre = nombre; // Error: this no existe
};
let p = new Persona("Ana"); // TypeError

// ✅ CORRECTO: Usa función tradicional para constructores
function Persona(nombre) {
    this.nombre = nombre;
}
let p = new Persona("Ana"); // Funciona

// ❌ ERROR: No puedes usar arrow function para métodos de objeto
let objeto = {
    valor: 42,
    duplicar: () => this.valor * 2 // this = window, no objeto
};

// ✅ CORRECTO: Usa función tradicional para métodos
let objeto = {
    valor: 42,
    duplicar: function() {
        return this.valor * 2; // this = objeto
    }
};

Comparación completa:

// Función declarada (tradicional)
function sumar(a, b) {
    return a + b;
}

// Función expresada (asignada)
let sumar = function(a, b) {
    return a + b;
};

// Arrow function (ES6)
let sumar = (a, b) => a + b;

// Todas hacen lo mismo, pero arrow function es más corta

¿Qué son los parámetros?

Los parámetros son variables que se definen en la declaración de la función. Los argumentos son los valores reales que se pasan cuando se llama la función.

// Argumentos: "Ana" y 25

function saludar(nombre = "Usuario") {

saludar();        // "Hola, Usuario!"
saludar("Juan");  // "Hola, Juan!"

Return - Valor de Retorno

¿Qué es return?

La palabra clave return especifica qué valor debe devolver la función. Una función sin return devuelve undefined.

Ejemplos:

// Función con return
function sumar(a, b) {

let resultado = sumar(5, 3); // resultado = 8

// Función sin return
function saludar(nombre) {
    console.log(`Hola, ${nombre}!`);

}

let saludo = saludar("Juan"); // saludo = undefined

Scope - Alcance de Variables

¿Qué es el scope?

El scope es el alcance o visibilidad de una variable. Define dónde puede ser accedida una variable en el código.

Tipos de scope:

  • Global: Accesible desde cualquier parte
  • Local: Solo accesible dentro de la función
  • Bloque: Solo accesible dentro del bloque {}

Ejemplo de scope local:

let global = "Soy global";

function ejemplo() {
    let local = "Soy local";
    console.log(global); // ✅ Accesible
    console.log(local);  // ✅ Accesible
}

ejemplo();
console.log(global); // ✅ Accesible
console.log(local);  // ❌ Error: local is not defined
×

Hoisting - Elevación

¿Qué es el hoisting?

El hoisting es el comportamiento de JavaScript de "elevar" las declaraciones de variables y funciones al inicio de su scope antes de la ejecución.

Hoisting de funciones:

// Esto funciona gracias al hoisting
console.log(saludar("Juan")); // ✅ "Hola Juan!"

function saludar(nombre) {
    return `Hola ${nombre}!`;
}

Hoisting de variables:

// var se eleva pero no se inicializa
console.log(x); // undefined (no error)
var x = 5;

// let/const se elevan pero no se pueden usar
console.log(y); // ❌ ReferenceError
let y = 10;

Métodos Básicos de Arrays

push() y pop()

Métodos para agregar y quitar elementos al final del array.

Sintaxis:

array.push(elemento1, elemento2, ...); // Agrega al final
let ultimo = array.pop(); // Quita el último elemento

Ejemplo:

let frutas = ["manzana", "banana"];

frutas.push("naranja"); // ["manzana", "banana", "naranja"]
let ultima = frutas.pop(); // "naranja"
// frutas ahora es ["manzana", "banana"]

shift() y unshift()

Métodos para agregar y quitar elementos al inicio del array.

let numeros = [2, 3, 4];

numeros.unshift(1); // [1, 2, 3, 4] - Agrega al inicio
let primero = numeros.shift(); // 1 - Quita el primero
// numeros ahora es [2, 3, 4]
×

Métodos de Transformación

map() - Transformar elementos

Crea un nuevo array aplicando una función a cada elemento.

let numeros = [1, 2, 3, 4, 5];

// Duplicar cada número
let dobles = numeros.map(num => num * 2);
console.log(dobles); // [2, 4, 6, 8, 10]

filter() - Filtrar elementos

Crea un nuevo array con elementos que cumplan una condición.

let edades = [15, 18, 25, 12, 30, 17];

// Solo mayores de edad
let mayores = edades.filter(edad => edad >= 18);
console.log(mayores); // [18, 25, 30]

reduce() - Reducir a un valor

Reduce el array a un solo valor aplicando una función acumuladora.

Los métodos son funciones que pertenecen a un objeto. Se definen como propiedades que contienen una función.

El destructuring es una forma de extraer valores de objetos y arrays de manera más concisa.

let usuario = {

let colores = ["rojo", "verde", "azul"];

Solución: Gestor de Lista de Compras

Explicación:

Este ejercicio combina arrays y objetos para crear un sistema de gestión. Aprenderás a agregar elementos, recorrer arrays y calcular totales.

Código de la Solución:

// Solución del Ejercicio 2: Gestor de Lista de Compras

// Array para almacenar productos
let listaCompras = [];

// Función para agregar producto
function agregarProducto(nombre, precio, cantidad) {
    let producto = {
        nombre: nombre,
        precio: precio,
        cantidad: cantidad,
        subtotal: precio * cantidad
    };
    listaCompras.push(producto);
    console.log(`${nombre} agregado a la lista`);
}

// Función para mostrar lista
function mostrarLista() {
    console.log("=== LISTA DE COMPRAS ===");
    for (let i = 0; i < listaCompras.length; i++) {
        let producto = listaCompras[i];
        console.log(`${i + 1}. ${producto.nombre} - $${producto.precio} x ${producto.cantidad} = $${producto.subtotal}`);
    }
}

// Función para calcular total
function calcularTotal() {
    let total = 0;
    for (let producto of listaCompras) {
        total += producto.subtotal;
    }
    return total;
}

// Ejemplo de uso
agregarProducto("Manzanas", 2.50, 3);
agregarProducto("Leche", 1.80, 2);
agregarProducto("Pan", 0.90, 1);

mostrarLista();
console.log("Total a pagar: $" + calcularTotal());

Puntos Clave:

  • Objetos: Cada producto es un objeto con propiedades
  • Arrays: La lista es un array que contiene objetos
  • push(): Agrega elementos al final del array
  • Bucles: for y for...of para recorrer arrays

Solución: Contador de Palabras

Explicación:

Este ejercicio te enseña a trabajar con strings y arrays usando métodos como split() y length. Es muy útil para procesar texto.

Código de la Solución:

// Solución del Ejercicio 4: Contador de Palabras

function contarPalabras(texto) {
    // Dividir el texto en palabras (separadas por espacios)
    let palabras = texto.split(' ');
    
    // Filtrar palabras vacías (espacios múltiples)
    let palabrasValidas = palabras.filter(palabra => palabra.length > 0);
    
    return palabrasValidas.length;
}

function contarCaracteres(texto) {
    return texto.length;
}

function encontrarPalabraMasLarga(texto) {
    let palabras = texto.split(' ').filter(palabra => palabra.length > 0);
    
    if (palabras.length === 0) {
        return "No hay palabras";
    }
    
    let palabraMasLarga = palabras[0];
    
    for (let palabra of palabras) {
        if (palabra.length > palabraMasLarga.length) {
            palabraMasLarga = palabra;
        }
    }
    
    return palabraMasLarga;
}

function analizarTexto(texto) {
    console.log("=== ANÁLISIS DE TEXTO ===");
    console.log("Texto original:", texto);
    console.log("Número de palabras:", contarPalabras(texto));
    console.log("Número de caracteres:", contarCaracteres(texto));
    console.log("Palabra más larga:", encontrarPalabraMasLarga(texto));
}

// Ejemplo de uso
let miTexto = "JavaScript es un lenguaje de programación muy poderoso";
analizarTexto(miTexto);

Puntos Clave:

  • split(): Divide un string en un array usando un separador
  • filter(): Crea un nuevo array con elementos que cumplen una condición
  • length: Propiedad que devuelve la longitud de strings y arrays
  • Bucles: for...of para recorrer arrays fácilmente
