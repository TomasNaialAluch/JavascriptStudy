Ejemplo Práctico

Operadores Básicos 
// Operadores aritméticos
let a = 10;
let b = 3;

console.log('Suma:', a + b);        // 13
console.log('Resta:', a - b);       // 7
console.log('Multiplicación:', a * b); // 30
console.log('División:', a / b);    // 3.33...
console.log('Módulo:', a % b);      // 1
console.log('Exponenciación:', a ** b); // 1000

// Operadores de comparación
let x = 5;
let y = "5";

console.log('Igualdad débil (==):', x == y);   // true
console.log('Igualdad estricta (===):', x === y); // false
console.log('Mayor que:', x > 3);    // true
console.log('Menor que:', x < 10);   // true

// Operadores lógicos
let esAdulto = true;
let tieneLicencia = false;

console.log('AND (&&):', esAdulto && tieneLicencia);  // false
console.log('OR (||):', esAdulto || tieneLicencia);   // true
console.log('NOT (!):', !esAdulto);                   // false
Precedencia y Type Coercion 
// Precedencia de operadores
console.log('Sin paréntesis:', 2 + 3 * 4);     // 14 (multiplicación primero)
console.log('Con paréntesis:', (2 + 3) * 4);   // 20

// Type coercion (conversión automática de tipos)
console.log('String + Number:', "5" + 3);      // "53" (concatenación)
console.log('Number + String:', 5 + "3");      // "53" (concatenación)
console.log('String - Number:', "5" - 3);      // 2 (resta convierte a número)
console.log('String * Number:', "5" * 3);      // 15 (multiplicación convierte a número)

// Comparaciones con type coercion
console.log('"5" == 5:', "5" == 5);           // true (conversión)
console.log('"5" === 5:', "5" === 5);         // false (sin conversión)
console.log('null == undefined:', null == undefined); // true
console.log('null === undefined:', null === undefined); // false

// Operadores de asignación
let numero = 10;
numero += 5;  // numero = numero + 5
console.log('+= 5:', numero);  // 15

numero *= 2;  // numero = numero * 2
console.log('*= 2:', numero);  // 30

// Operadores de incremento/decremento
let contador = 5;
console.log('Pre-incremento:', ++contador);  // 6
console.log('Post-incremento:', contador++); // 6 (pero contador ahora es 7)
console.log('Valor final:', contador);       // 7

Ejemplo Práctico

Condicionales y Bucles Básicos 
// Condicionales
let temperatura = 25;

if (temperatura > 30) {
    console.log("Hace mucho calor");
} else if (temperatura > 20) {
    console.log("Temperatura agradable");
} else if (temperatura > 10) {
    console.log("Hace fresco");
} else {
    console.log("Hace frío");
}

// Operador ternario
let mensaje = temperatura > 25 ? "Calor" : "Fresco";
console.log("Mensaje:", mensaje);

// Switch
let dia = "lunes";

switch (dia) {
    case "lunes":
        console.log("Inicio de semana");
        break;
    case "viernes":
        console.log("¡Fin de semana!");
        break;
    case "sábado":
    case "domingo":
        console.log("Día de descanso");
        break;
    default:
        console.log("Día laboral");
}

// Bucles básicos
console.log("--- For Loop ---");
for (let i = 0; i < 5; i++) {
    console.log(`Iteración ${i}`);
}

console.log("--- While Loop ---");
let contador = 0;
while (contador < 3) {
    console.log(`Contador: ${contador}`);
    contador++;
}
Iteración de Arrays y Objetos 
// Arrays
let frutas = ["manzana", "banana", "naranja", "uva"];

console.log("--- For...of (valores) ---");
for (let fruta of frutas) {
    console.log(fruta);
}

console.log("--- For tradicional con índices ---");
for (let i = 0; i < frutas.length; i++) {
    console.log(`${i}: ${frutas[i]}`);
}

// Objetos
let persona = { 
    nombre: "Juan", 
    edad: 30, 
    ciudad: "Madrid",
    profesion: "Desarrollador"
};

console.log("--- For...in (propiedades) ---");
for (let propiedad in persona) {
    console.log(`${propiedad}: ${persona[propiedad]}`);
}

// Break y Continue
console.log("--- Break y Continue ---");
for (let i = 0; i < 10; i++) {
    if (i === 3) {
        console.log("Saltando 3...");
        continue; // Salta esta iteración
    }
    if (i === 7) {
        console.log("Llegamos a 7, terminando...");
        break; // Sale del bucle
    }
    console.log(`Número: ${i}`);
}

// Bucle anidado
console.log("--- Bucle Anidado ---");
for (let i = 1; i <= 3; i++) {
    for (let j = 1; j <= 2; j++) {
        console.log(`i=${i}, j=${j}`);
    }
}

Ejemplo Práctico

Declaración Básica de Funciones 
// Función declarada - se puede usar antes de declararla
function saludar(nombre) {
    return `Hola, ${nombre}!`;
}

// Función expresada - se debe declarar antes de usar
let sumar = function(a, b) {
    return a + b;
};

// Arrow function (ES6) - más corta y moderna
let multiplicar = (a, b) => a * b;

// Uso de las funciones
console.log(saludar("María"));           // "Hola, María!"
console.log(sumar(5, 3));                // 8
console.log(multiplicar(4, 6));          // 24

// Función con parámetros por defecto
function crearUsuario(nombre, edad = 18, activo = true) {
    return {
        nombre,
        edad,
        activo
    };
}

console.log(crearUsuario("Juan", 25));   // { nombre: "Juan", edad: 25, activo: true }
console.log(crearUsuario("Ana"));        // { nombre: "Ana", edad: 18, activo: true }
Scope, Hoisting y Closures 
// HOISTING - Las funciones declaradas se "elevan"
console.log(saludarHoisted("Juan")); // ✅ Funciona - "Hola Juan!"

function saludarHoisted(nombre) {
    return `Hola ${nombre}!`;
}

// ❌ Esto NO funciona con funciones expresadas
// console.log(saludarExpressed("Ana")); // ReferenceError

let saludarExpressed = function(nombre) {
    return `Hola ${nombre}!`;
};

// SCOPE - Alcance de variables
let variableGlobal = "Soy global";

function ejemploScope() {
    let variableLocal = "Soy local";
    var variableVar = "Soy var";
    
    console.log("Dentro de la función:");
    console.log("Global:", variableGlobal);  // ✅ Accesible
    console.log("Local:", variableLocal);    // ✅ Accesible
    console.log("Var:", variableVar);        // ✅ Accesible
}

ejemploScope();

// ❌ Esto NO funciona - variableLocal no existe fuera
// console.log(variableLocal); // ReferenceError

// CLOSURE - Función que "recuerda" su entorno
function crearContador() {
    let contador = 0;
    
    return function() {
        contador++;
        return contador;
    };
}

let miContador = crearContador();
console.log(miContador()); // 1
console.log(miContador()); // 2
console.log(miContador()); // 3

// Cada contador es independiente
let otroContador = crearContador();
console.log(otroContador()); // 1 (no 4)

let frutas = ["manzana", "banana", "naranja"];

// MÉTODOS DE TRANSFORMACIÓN DE ARRAYS
let numeros = [1, 2, 3, 4, 5];

// map() - Transformar cada elemento

console.log("Dobles:", dobles); // [2, 4, 6, 8, 10]

// filter() - Filtrar elementos
let pares = numeros.filter(num => num % 2 === 0);
console.log("Pares:", pares); // [2, 4]

// reduce() - Reducir a un solo valor
let suma = numeros.reduce((acumulador, num) => acumulador + num, 0);
console.log("Suma total:", suma); // 15

// BÚSQUEDA EN ARRAYS

// Solución del Ejercicio 2: Gestor de Lista de Compras

// Array para almacenar productos
let listaCompras = [];

// Función para agregar producto
function agregarProducto(nombre, precio, cantidad = 1) {
    let producto = {
        nombre: nombre,
        precio: precio,
        cantidad: cantidad,
        total: precio * cantidad
    };
    
    listaCompras.push(producto);
    console.log(`✅ ${nombre} agregado a la lista`);
}

// Función para mostrar la lista
function mostrarLista() {
    console.log("🛒 Lista de Compras:");
    console.log("==================");
    
    for (let i = 0; i < listaCompras.length; i++) {
        let producto = listaCompras[i];
        console.log(`${i + 1}. ${producto.nombre} - $${producto.precio} x ${producto.cantidad} = $${producto.total}`);
    }
}

// Función para calcular total
function calcularTotal() {
    let total = 0;
    
    for (let producto of listaCompras) {
        total += producto.total;
    }
    
    return total;
}

// Probar el sistema

// Solución del Ejercicio 4: Contador de Palabras

function contarPalabras(texto) {
    // Dividir el texto en palabras (separadas por espacios)
    let palabras = texto.split(' ');
    
    // Filtrar palabras vacías (por si hay espacios dobles)
    let palabrasReales = palabras.filter(palabra => palabra.length > 0);
    
    return palabrasReales.length;
}

function contarCaracteres(texto) {
    return texto.length;
}

function palabraMasLarga(texto) {
    let palabras = texto.split(' ');
    let palabraLarga = '';
    
    for (let palabra of palabras) {
        if (palabra.length > palabraLarga.length) {
            palabraLarga = palabra;
        }
    }
    
    return palabraLarga;
}

function analizarTexto(texto) {
    console.log("📝 Análisis del texto:");

if/else - Condicionales

×

¿Qué es if/else?

Es la estructura de control más básica que permite que tu código tome decisiones. Evalúa una condición y ejecuta diferentes bloques de código según el resultado.

Sintaxis:

if (condición) {
    // Código si la condición es verdadera
} else if (otra_condición) {
    // Código si la segunda condición es verdadera
} else {
    // Código si ninguna condición es verdadera
}

Ejemplo Práctico:

let edad = 18;

if (edad >= 18) {
    console.log("Eres mayor de edad");
} else if (edad >= 13) {
    console.log("Eres adolescente");
} else {
    console.log("Eres menor de edad");
}

Puntos Clave:

  • La condición debe evaluar a true o false
  • Puedes tener múltiples else if
  • El else es opcional
  • Usa {} para bloques de múltiples líneas

switch/case - Múltiples Casos

×

¿Qué es switch y case?

switch es una estructura que evalúa una expresión y ejecuta diferentes bloques de código según el valor. case define cada posible valor y qué hacer en cada caso.

Sintaxis:

switch (expresión) {
    case valor1:
        // Código para valor1
        break;
    case valor2:
        // Código para valor2
        break;
    default:
        // Código si no coincide con ningún caso
        break;
}

¿Qué hace cada parte?

  • switch (expresión): Evalúa la expresión una sola vez
  • case valor: Compara el valor con la expresión
  • break: Sale del switch (¡MUY IMPORTANTE!)
  • default: Se ejecuta si ningún case coincide

Ejemplo Práctico:

let dia = "lunes";

switch (dia) {
    case "lunes":
        console.log("Inicio de semana");
        break;
    case "viernes":
        console.log("¡Fin de semana!");
        break;
    case "sábado":
    case "domingo":
        console.log("Día de descanso");
        break;
    default:
        console.log("Día laboral");
}

⚠️ ¡Cuidado con el break!

Sin break, el código "cae" al siguiente case (fall-through). Esto puede ser útil o causar errores.

for - Bucle con Contador

×

¿Qué es el bucle for?

Es un bucle que se repite un número específico de veces usando un contador. Es perfecto cuando sabes exactamente cuántas veces quieres que se ejecute el código.

Sintaxis:

for (inicialización; condición; actualización) {
    // Código que se repite
}

¿Qué hace cada parte?

  • inicialización: Se ejecuta UNA VEZ al inicio (ej: let i = 0)
  • condición: Se evalúa ANTES de cada iteración (ej: i < 5)
  • actualización: Se ejecuta DESPUÉS de cada iteración (ej: i++)

Ejemplo Paso a Paso:

for (let i = 0; i < 3; i++) {
    console.log(`Iteración ${i}`);
}

// Resultado:
// Iteración 0
// Iteración 1
// Iteración 2

Proceso paso a paso:

  1. Inicio: i = 0
  2. ¿i < 3? Sí (0 < 3) → Ejecuta código → i++ → i = 1
  3. ¿i < 3? Sí (1 < 3) → Ejecuta código → i++ → i = 2
  4. ¿i < 3? Sí (2 < 3) → Ejecuta código → i++ → i = 3
  5. ¿i < 3? No (3 < 3) → Termina el bucle

Usos Comunes:

  • Iterar arrays con índices: for (let i = 0; i < array.length; i++)
  • Repetir código N veces
  • Generar secuencias numéricas

for...of - Iterar VALORES

×

¿Qué es for...of?

Es un bucle que itera sobre los VALORES de elementos iterables (arrays, strings, etc.). Es más simple que el for tradicional cuando solo necesitas los valores, no los índices.

Sintaxis:

for (let elemento of iterable) {
    // Código que usa el elemento
}

Ejemplo con Array:

let frutas = ["manzana", "banana", "naranja"];

for (let fruta of frutas) {
    console.log(fruta);
}

// Resultado:
// manzana
// banana
// naranja

Comparación: for vs for...of

for tradicional (con índices):

for (let i = 0; i < frutas.length; i++) {
    console.log(`${i}: ${frutas[i]}`);
}
// Resultado:
// 0: manzana
// 1: banana
// 2: naranja

for...of (solo valores):

for (let fruta of frutas) {
    console.log(fruta);
}
// Resultado:
// manzana
// banana
// naranja

¿Cuándo usar for...of?

  • ✅ Cuando solo necesitas los valores
  • ✅ Código más limpio y legible
  • ✅ Menos propenso a errores
  • ❌ NO cuando necesitas el índice
  • ❌ NO para modificar el array original

for...in - Iterar PROPIEDADES

×

¿Qué es for...in?

Es un bucle que itera sobre las PROPIEDADES (claves) de un objeto. Te da acceso a los nombres de las propiedades, no a sus valores.

Sintaxis:

for (let propiedad in objeto) {
    // Código que usa la propiedad
}

Ejemplo con Objeto:

let persona = {
    nombre: "Juan",
    edad: 25,
    ciudad: "Madrid"
};

for (let propiedad in persona) {
    console.log(`${propiedad}: ${persona[propiedad]}`);
}

// Resultado:
// nombre: Juan
// edad: 25
// ciudad: Madrid

¿Cómo acceder a los valores?

Usa la notación de corchetes: objeto[propiedad]

for (let prop in persona) {
    console.log(prop);                    // nombre, edad, ciudad
    console.log(persona[prop]);          // Juan, 25, Madrid
    console.log(persona.prop);           // undefined (¡ERROR!)
}

Diferencia clave: for...of vs for...in

for...of (valores):

let array = [10, 20, 30];
for (let valor of array) {
    console.log(valor);  // 10, 20, 30
}

for...in (propiedades):

let array = [10, 20, 30];
for (let indice in array) {
    console.log(indice);  // 0, 1, 2
    console.log(array[indice]); // 10, 20, 30
}

⚠️ Advertencias importantes:

  • for...in también itera propiedades heredadas
  • El orden no está garantizado en objetos
  • Para arrays, usa for...of o for tradicional

while - Mientras se cumple condición

×

¿Qué es el bucle while?

Es un bucle que se repite mientras una condición sea verdadera. No sabes cuántas veces se ejecutará, solo sabes cuándo debe parar.

Sintaxis:

while (condición) {
    // Código que se repite
    // ¡IMPORTANTE! Modificar la condición aquí
}

Ejemplo Básico:

let contador = 0;

while (contador < 3) {
    console.log(`Contador: ${contador}`);
    contador++;  // ¡IMPORTANTE! Sin esto sería un bucle infinito
}

// Resultado:
// Contador: 0
// Contador: 1
// Contador: 2

Proceso paso a paso:

  1. ¿contador < 3? Sí (0 < 3) → Ejecuta → contador = 1
  2. ¿contador < 3? Sí (1 < 3) → Ejecuta → contador = 2
  3. ¿contador < 3? Sí (2 < 3) → Ejecuta → contador = 3
  4. ¿contador < 3? No (3 < 3) → Termina

⚠️ ¡Cuidado con los bucles infinitos!

// ❌ MAL - Bucle infinito
let i = 0;
while (i < 5) {
    console.log("Hola");
    // ¡Falta i++!
}

// ✅ BIEN - Condición que cambia
let i = 0;
while (i < 5) {
    console.log("Hola");
    i++;  // La condición cambia
}

¿Cuándo usar while?

  • ✅ Cuando no sabes cuántas iteraciones necesitas
  • ✅ Validación de entrada del usuario
  • ✅ Procesar datos hasta encontrar un valor específico
  • ❌ NO cuando sabes exactamente cuántas veces iterar (usa for)

Operador Ternario (? :)

×

¿Qué es el operador ternario?

Es una forma abreviada de escribir un if/else simple en una sola línea. Es perfecto para asignaciones condicionales.

Sintaxis:

condición ? valor_si_verdadero : valor_si_falso

Comparación: if/else vs Ternario

if/else tradicional:

let edad = 18;
let mensaje;

if (edad >= 18) {
    mensaje = "Mayor de edad";
} else {
    mensaje = "Menor de edad";
}

Operador ternario:

let edad = 18;
let mensaje = edad >= 18 ? "Mayor de edad" : "Menor de edad";

Ejemplos Prácticos:

// Asignación condicional
let precio = 100;
let descuento = precio > 50 ? 10 : 0;

// En template strings
let nombre = "Juan";
let saludo = `Hola ${nombre}, ${nombre.length > 3 ? "bienvenido" : "hola"}`;

// Retorno de función
function esPar(numero) {
    return numero % 2 === 0 ? "par" : "impar";
}

⚠️ No abuses del ternario:

// ❌ MAL - Muy difícil de leer
let resultado = a > b ? c > d ? "A" : "B" : c > d ? "C" : "D";

// ✅ BIEN - Usa if/else para casos complejos
let resultado;
if (a > b) {
    resultado = c > d ? "A" : "B";
} else {
    resultado = c > d ? "C" : "D";
}

¿Cuándo usar ternario?

  • ✅ Asignaciones simples (una condición)
  • ✅ Valores por defecto
  • ✅ Template strings
  • ❌ NO para lógica compleja
  • ❌ NO para múltiples condiciones

Función Declarada

×

¿Qué es una función declarada?

Es la forma tradicional de crear funciones en JavaScript. Se declara con la palabra clave function seguida del nombre de la función.

Sintaxis:

function nombreFuncion(parametros) {
    // código a ejecutar
    return valor;
}

Características:

  • Hoisting: Se puede usar antes de declararla
  • Nombre obligatorio: Debe tener un nombre
  • Scope global: Se crea en el scope donde se declara

Ejemplo:

// Se puede usar antes de declararla
console.log(saludar("Juan")); // ✅ Funciona

function saludar(nombre) {
    return `Hola, ${nombre}!`;
}

Función Expresada

×

¿Qué es una función expresada?

Es una función que se asigna a una variable. Se crea usando una expresión de función anónima (sin nombre propio).

Sintaxis:

let nombreVariable = function(parametros) {
    // código a ejecutar
    return valor;
};

¿Cuándo se usa?

  • Funciones condicionales: Cuando quieres crear diferentes funciones según una condición
  • Callbacks: Para pasar funciones como parámetros a otras funciones
  • Funciones dinámicas: Cuando el nombre de la función puede cambiar
  • Módulos: Para exportar funciones específicas

Ejemplos prácticos:

// 1. Función condicional
let operacion;
if (usuario.esAdmin) {
    operacion = function() {
        return "Acceso completo";
    };
} else {
    operacion = function() {
        return "Acceso limitado";
    };
}

// 2. Callback en setTimeout
setTimeout(function() {
    console.log("Se ejecutó después de 2 segundos");
}, 2000);

// 3. Función como parámetro
function procesarDatos(datos, callback) {
    let resultado = datos * 2;
    callback(resultado);
}

let miCallback = function(resultado) {
    console.log("El resultado es:", resultado);
};

procesarDatos(5, miCallback); // "El resultado es: 10"

Características:

  • Sin hoisting: No se puede usar antes de declararla
  • Asignada a variable: Se almacena en una variable
  • Anónima: La función no tiene nombre propio
  • Flexible: Se puede reasignar o pasar como parámetro

Comparación con función declarada:

// ❌ Esto NO funciona - función expresada
console.log(saludarExpresada("Juan")); // ReferenceError

let saludarExpresada = function(nombre) {
    return `Hola, ${nombre}!`;
};

// ✅ Esto SÍ funciona - función declarada
console.log(saludarDeclarada("Juan")); // "Hola, Juan!"

function saludarDeclarada(nombre) {
    return `Hola, ${nombre}!`;

Arrow Function (ES6)

×

¿Qué es ES6?

ES6 (ECMAScript 2015) es una versión importante de JavaScript que se lanzó en 2015. Introdujo muchas características modernas que hicieron JavaScript más potente y fácil de usar. ES6 incluye:

  • Arrow functions: Sintaxis más corta para funciones
  • let y const: Nuevas formas de declarar variables
  • Template literals: Strings con interpolación (backticks)
  • Destructuring: Extraer valores de objetos y arrays
  • Classes: Programación orientada a objetos
  • Modules: Importar y exportar código

¿Qué es una arrow function?

Es una forma más corta y moderna de escribir funciones, introducida en ES6. Usa la sintaxis => (flecha) y es muy popular en JavaScript moderno.

Sintaxis:

// Con un parámetro
let nombre = parametro => expresion;

// Con múltiples parámetros
let nombre = (param1, param2) => expresion;

// Con bloque de código
let nombre = (param1, param2) => {
    // código
    return valor;
};

¿Cuándo usar arrow functions?

  • Callbacks cortos: En map(), filter(), forEach()
  • Funciones simples: Cuando la lógica es básica
  • Event handlers: Para manejar eventos del DOM
  • Promesas: En .then() y .catch()

Ejemplos prácticos:

// 1. Con arrays (muy común)
let numeros = [1, 2, 3, 4, 5];

// Función tradicional
let dobles = numeros.map(function(num) {
    return num * 2;
});

// Arrow function (más corta)
let dobles = numeros.map(num => num * 2);

// 2. Con eventos
document.addEventListener('click', () => {
    console.log('Se hizo clic en la página');
});

// 3. Con promesas
fetch('/api/datos')
    .then(response => response.json())
    .then(data => console.log(data));

Características importantes:

  • Más corta: Sintaxis más concisa y legible
  • Sin this propio: Hereda el this del contexto padre
  • Return implícito: Si es una expresión, no necesita return
  • No se puede usar como constructor: No se puede usar con 'new'

Explicación Detallada de Características:

1. Sintaxis más concisa:

Las arrow functions reducen significativamente la cantidad de código necesario:

// Función tradicional - 3 líneas
let numeros = [1, 2, 3, 4, 5];
let dobles = numeros.map(function(numero) {
    return numero * 2;
});

// Arrow function - 1 línea
let dobles = numeros.map(numero => numero * 2);

// Ahorro: 66% menos código

2. Sin this propio (muy importante):

Las arrow functions NO crean su propio contexto de this, heredan el del contexto padre:

let persona = {
    nombre: "Ana",
    edad: 25,
    
    // ❌ PROBLEMA con función tradicional
    saludarTradicional: function() {
        setTimeout(function() {
            console.log(`Hola, soy ${this.nombre}`); // this = undefined
        }, 1000);
    },
    
    // ✅ SOLUCIÓN con arrow function
    saludarArrow: function() {
        setTimeout(() => {
            console.log(`Hola, soy ${this.nombre}`); // this = persona
        }, 1000);
    }
};

persona.saludarTradicional(); // "Hola, soy undefined"
persona.saludarArrow();       // "Hola, soy Ana"

3. Return implícito (cuando NO usar llaves):

Si la función es una sola expresión, NO necesitas return ni llaves:

// ✅ CON return implícito (sin llaves)
let sumar = (a, b) => a + b;
let esPar = num => num % 2 === 0;
let obtenerNombre = usuario => usuario.nombre;

// ❌ INCORRECTO - no necesitas return aquí
let sumar = (a, b) => { return a + b; };

// ✅ CON return explícito (con llaves)
let procesarDatos = (datos) => {
    let resultado = datos * 2;
    console.log("Procesando...");
    return resultado; // Necesario porque hay más código
};

// ✅ CON return implícito (una sola expresión)
let procesarDatos = (datos) => datos * 2;

Casos de uso reales:

// 1. Con arrays (muy común)
let precios = [10, 20, 30, 40];
let preciosConIva = precios.map(precio => precio * 1.21);
let preciosAltos = precios.filter(precio => precio > 25);

// 2. Con eventos del DOM
document.getElementById('boton').addEventListener('click', () => {
    console.log('Botón presionado');
});

// 3. Con promesas
fetch('/api/datos')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

Errores comunes a evitar:

// ❌ ERROR: No puedes usar arrow function como constructor
let Persona = (nombre) => {
    this.nombre = nombre; // Error: this no existe
};
let p = new Persona("Ana"); // TypeError

// ✅ CORRECTO: Usa función tradicional para constructores
function Persona(nombre) {
    this.nombre = nombre;
}
let p = new Persona("Ana"); // Funciona

// ❌ ERROR: No puedes usar arrow function para métodos de objeto
let objeto = {
    valor: 42,
    duplicar: () => this.valor * 2 // this = window, no objeto
};

// ✅ CORRECTO: Usa función tradicional para métodos
let objeto = {
    valor: 42,
    duplicar: function() {
        return this.valor * 2; // this = objeto
    }
};

Comparación completa:

// Función declarada (tradicional)
function sumar(a, b) {
    return a + b;
}

// Función expresada (asignada)
let sumar = function(a, b) {
    return a + b;
};

// Arrow function (ES6)
let sumar = (a, b) => a + b;

// Todas hacen lo mismo, pero arrow function es más corta

Return - Valor de Retorno

¿Qué es var?

var es la forma tradicional (pre-ES6) de declarar variables en JavaScript. Aunque sigue funcionando, tiene peculiaridades que pueden causar errores.

Características de var:

1. Function Scope (Alcance de Función):

function ejemplo() {

console.log(nombre); // undefined (no error)

nombre = "Juan";         // Asignación queda en su lugar

3. Re-declaración Permitida:

var edad = 25;

// Problema 3: Re-declaración confusa

// ... muchas líneas después ...

  • ❌ En código nuevo (usa let o const)

  • ❌ Para evitar confusiones con hoisting

Cuándo podrías usar var:

  • ✅ Manteniendo código legacy
  • ✅ Compatibilidad con navegadores muy antiguos

¿Qué es let?

let es la forma moderna (ES6+) de declarar variables. Tiene block scope y soluciona muchos problemas de var.

Sintaxis:

Características de let:

1. Block Scope (Alcance de Bloque):

function ejemplo() {
    if (true) {
        let x = 10;
        console.log(x); // 10 ✅
    }
    console.log(x); // ❌ Error: x is not defined
}

// El scope es el bloque {...}
{
    let mensaje = "Hola";

}
console.log(mensaje); // ❌ Error: mensaje is not defined

let edad = 25;
let edad = 30; // ❌ Error: Identifier 'edad' has already been declared

// Pero SÍ se puede reasignar:
edad = 30; // ✅ Correcto
console.log(edad); // 30

3. Temporal Dead Zone (TDZ):

console.log(nombre); // ❌ Error: Cannot access 'nombre' before initialization
let nombre = "Juan";

// let SÍ tiene hoisting, pero no se puede acceder antes de la declaración

4. Perfecto para Loops:

// Con let - cada iteración tiene su propio scope
for (let i = 0; i < 3; i++) {
    setTimeout(function() {
        console.log(i); // Imprime 0, 1, 2 ✅
    }, 100);
}

// Comparación con var:
for (var j = 0; j < 3; j++) {
    setTimeout(function() {
        console.log(j); // Imprime 3, 3, 3 ❌
    }, 100);
}

Tipos de Números:

let infinito = Infinity;
let menosInfinito = -Infinity;
let noEsNumero = NaN; // Not a Number

console.log(typeof NaN); // "number"

Operaciones Matemáticas:

let a = 10;

// Valor absoluto

// Máximo y mínimo

Un boolean representa un valor lógico que puede ser true (verdadero) o false (falso).

let verdadero = true;

console.log(Boolean(false));     // false

console.log(Boolean(-0));        // false
console.log(Boolean(0n));        // false (BigInt)
console.log(Boolean(""));        // false (string vacío)
console.log(Boolean(null));      // false
console.log(Boolean(undefined)); // false
console.log(Boolean(NaN));       // false

Valores Truthy (Verdaderos):

// Casi todo lo demás es true:
console.log(Boolean(true));      // true
console.log(Boolean(1));         // true
console.log(Boolean(-1));        // true
console.log(Boolean("0"));       // true (string con "0")
console.log(Boolean("false"));   // true (string con "false")
console.log(Boolean([]));        // true (array vacío)
console.log(Boolean({}));        // true (objeto vacío)
console.log(Boolean(function(){})); // true (función)

Operadores Lógicos:

1. AND (&&):

// Devuelve el primer valor falsy o el último si todos son truthy
console.log(true && true);   // true
console.log(true && false);  // false
console.log(false && true);  // false
console.log(false && false); // false

// Ejemplos prácticos
let edad = 25;
let tieneLicencia = true;
let puedeConducir = edad >= 18 && tieneLicencia; // true

// Short-circuit evaluation
let usuario = null;

console.log(false || true);  // true

// Valores por defecto

// Configuración

3. NOT (!):

console.log(!false); // true

// Doble negación para convertir a boolean

¿Qué es un Callback?

×

Definición:

Un callback es una función que se pasa como argumento a otra función para ser ejecutada más tarde.

Concepto Básico:

// Función que recibe un callback
function hacerAlgo(callback) {
    console.log("Haciendo algo...");
    callback(); // Ejecutar el callback
}

// Función que usaremos como callback
function terminado() {
    console.log("¡Terminado!");
}

// Pasar la función como argumento
hacerAlgo(terminado);
// Output:
// Haciendo algo...
// ¡Terminado!

Por qué se llama "callback":

"Call back" significa "llamar de vuelta". La función principal "llama de vuelta" a la función que le pasaste cuando está lista para ejecutarla.

Puntos Clave:

  • Función como valor: En JavaScript, las funciones son valores que se pueden pasar
  • Ejecución diferida: El callback se ejecuta cuando la función que lo recibe decide
  • Muy común: Se usa en eventos, arrays, operaciones asíncronas, etc.

  • Para personalizar el comportamiento de una función
  • Para operaciones asíncronas (que toman tiempo)
  • Para eventos del navegador (clicks, teclas, etc.)

setTimeout()

¿Qué es setTimeout?

setTimeout() ejecuta una función después de un tiempo específico (en milisegundos).

Sintaxis:

setTimeout(callback, tiempoEnMilisegundos);

Ejemplo Básico:

console.log("Inicio");

setTimeout(function() {
    console.log("Esto se ejecuta después de 2 segundos");
}, 2000);

console.log("Fin");

// Output:
// Inicio
// Fin
// Esto se ejecuta después de 2 segundos (2 segundos más tarde)

Con Arrow Function:

setTimeout(() => {
    console.log("¡Hola desde el futuro!");
}, 3000);

Pasar Parámetros:

function saludar(nombre, edad) {
    console.log(`Hola ${nombre}, tienes ${edad} años`);
}

// Los parámetros adicionales se pasan después del tiempo
setTimeout(saludar, 1000, "Juan", 25);
// Output (después de 1 segundo): Hola Juan, tienes 25 años

Cancelar un setTimeout:

// Guardar el ID del timeout
let timeoutId = setTimeout(() => {
    console.log("Esto no se ejecutará");
}, 5000);

// Cancelar antes de que se ejecute
clearTimeout(timeoutId);
console.log("Timeout cancelado");

Casos de Uso Reales:

  • Notificaciones: Mostrar un mensaje después de cierto tiempo
  • Animaciones: Ejecutar animaciones con delay
  • Polling: Verificar algo periódicamente
  • Debouncing: Esperar a que el usuario termine de escribir

⚠️ Importante:

El código después de setTimeout se ejecuta inmediatamente, NO espera. Es asíncrono.

setInterval()

¿Qué es setInterval?

setInterval() ejecuta una función repetidamente cada cierto tiempo.

Sintaxis:

setInterval(callback, tiempoEnMilisegundos);

Ejemplo Básico:

let contador = 0;

setInterval(function() {
    contador++;
    console.log(`Contador: ${contador}`);
}, 1000);

// Output (cada segundo):
// Contador: 1
// Contador: 2
// Contador: 3
// ... (continúa indefinidamente)

Detener un Interval:

let contador = 0;

let intervalId = setInterval(() => {
    contador++;
    console.log(`Contador: ${contador}`);
    
    // Detener después de 5 veces
    if (contador >= 5) {
        clearInterval(intervalId);
        console.log("Interval detenido");
    }

Síncrono vs Asíncrono

Código Síncrono:

Se ejecuta línea por línea, esperando que cada operación termine antes de continuar.

console.log("1. Primero");
console.log("2. Segundo");
console.log("3. Tercero");

// Output (en orden):
// 1. Primero
// 2. Segundo
// 3. Tercero

Código Asíncrono:

No espera a que una operación termine. Continúa ejecutando el siguiente código.

console.log("1. Primero");

setTimeout(() => {
    console.log("2. Segundo (asíncrono)");
}, 1000);

console.log("3. Tercero");

// Output:
// 1. Primero
// 3. Tercero
// 2. Segundo (asíncrono) ← Aparece 1 segundo después

Comparación Visual:

// SÍNCRONO: Espera en cada paso
function tarea1() {
    console.log("Tarea 1 iniciada");
    // Código que toma 2 segundos
    console.log("Tarea 1 completada");
}

function tarea2() {
    console.log("Tarea 2 iniciada");
    console.log("Tarea 2 completada");
}

tarea1(); // Espera que termine
tarea2(); // Luego ejecuta esto

// VS

// ASÍNCRONO: No espera
function tarea1Async() {
    console.log("Tarea 1 iniciada");
    setTimeout(() => {
        console.log("Tarea 1 completada");
    }, 2000);
}

function tarea2() {
    console.log("Tarea 2 iniciada");
    console.log("Tarea 2 completada");
}

tarea1Async(); // No espera, continúa inmediatamente
tarea2(); // Se ejecuta mientras tarea1 está en proceso

Operaciones Asíncronas Comunes:

  • setTimeout / setInterval: Temporizadores
  • Eventos: Clicks, teclas, etc.
  • Solicitudes HTTP: Obtener datos de un servidor
  • Lectura de archivos: Cargar archivos
  • Animaciones: requestAnimationFrame

¿Por qué necesitamos código asíncrono?

  • No bloquear: El navegador puede seguir respondiendo
  • Múltiples tareas: Hacer varias cosas a la vez
  • Mejor experiencia: La aplicación no se congela
  • Eficiencia: No esperar innecesariamente

Ejemplo Práctico:

// Mientras tanto, veo TV

forEach()

¿Qué es forEach?

forEach() es un método de arrays que ejecuta un callback para cada elemento.

Sintaxis:

array.forEach(function(elemento, indice, arrayCompleto) {
    // Código a ejecutar por cada elemento
});

Ejemplo Básico:

let frutas = ['manzana', 'banana', 'naranja'];

frutas.forEach(function(fruta) {
    console.log(fruta);
});

// manzana
// banana
// naranja

    console.log(`${indice}: ${color}`);
});

// Output:
// 0: rojo
// 1: verde
// 2: azul

Con Arrow Function:

let numeros = [1, 2, 3, 4, 5];

numeros.forEach(numero => {
    console.log(numero * 2);
});

// Output: 2, 4, 6, 8, 10

Modificar Elementos del DOM:

// HTML: 
// <ul id="lista"></ul>

let tareas = ['Estudiar', 'Ejercicio', 'Leer'];

// - Estudiar
// - Ejercicio
// - Leer

// María tiene 21 años

  • Modificar el DOM: Crear elementos para cada item

⚠️ Cuándo NO usar forEach:

}

    this en Callbacks

    🤔 ¿Qué es "this"? - Explicación Simple

    Imagina que this es como la palabra "yo" o "mi" cuando hablas.

    Ejemplo del mundo real:

    • Cuando tú dices "mi nombre", te refieres a TU nombre
    • Cuando yo digo "mi nombre", me refiero a MI nombre
    • La misma frase ("mi nombre"), pero depende de quién la dice

    En JavaScript, this funciona igual: siempre se refiere a "quién está ejecutando este código".

    📋 Las 4 Reglas Básicas de "this"

    JavaScript decide qué es this usando 4 reglas simples:

    Regla 1: Llamada de Método (lo más común)

    Si llamas una función así: objeto.metodo()
    Entonces this es: el objeto

    let persona = {
    nombre: "Ana",
    saludar: function() {
        console.log(this.nombre); // "this" es "persona"
    }
};

persona.saludar(); // "Ana" ← persona está ejecutando el código

    Regla 2: Llamada Simple (función suelta)

    Si llamas una función así: funcion()
    Entonces this es: window (o undefined en modo estricto)

    function decirHola() {
    console.log(this); // window o undefined
}

decirHola(); // No hay objeto, entonces "this" es window

    Regla 3: Constructor (con "new")

    Si llamas con "new": new Funcion()
    Entonces this es: el nuevo objeto que se crea

    function Persona(nombre) {
    this.nombre = nombre; // "this" es el nuevo objeto
}

let ana = new Persona("Ana"); // "this" es el objeto "ana"

    Regla 4: Forzar "this" manualmente

    Con call/apply/bind: Tú decides qué es this

    function saludar() {
    console.log(`Hola, soy ${this.nombre}`);
}

let persona = { nombre: "Carlos" };
saludar.call(persona); // Forzamos que "this" sea "persona"

    🎯 El Problema con Callbacks - Explicación Intuitiva

    Piensa en esto como delegar una tarea:

    • Es como si le das tu teléfono a un amigo para que haga una llamada
    • Tu amigo hace la llamada, pero él no sabe quién eres tú
    • Cuando alguien pregunta "¿quién llama?", tu amigo no puede responder por ti

    Ejemplo del Problema:

    let persona = {
    nombre: "Ana",
    edad: 25,
    
    // Método 1: Funciona bien
    saludar: function() {
        console.log(`Hola, soy ${this.nombre}`);
        // ✅ "this" es "persona" porque persona.saludar() (Regla 1)
    },
    
    // Método 2: NO funciona
    saludarDespues: function() {
        setTimeout(function() {
            // ❌ Aquí "this" NO es "persona"
            // Porque setTimeout llama la función, no "persona"
            console.log(`Hola, soy ${this.nombre}`);
        }, 1000);
    }
};

persona.saludar();         // ✅ "Hola, soy Ana"
persona.saludarDespues();  // ❌ "Hola, soy undefined"

    🔍 ¿Por qué pasa esto? - Paso a Paso

    Vamos a ver exactamente qué pasa en cada paso:

    let persona = {
    nombre: "Ana",
    saludarDespues: function() {
        // PASO 1: Aquí estamos DENTRO de persona.saludarDespues()
        // "this" = persona ✅
        
        setTimeout(function() {
            // PASO 2: Esta función se ejecuta 1 segundo DESPUÉS
            // ¿Quién la ejecuta? setTimeout (no persona)
            // Entonces "this" ya NO es persona ❌
            // "this" = window (o undefined)
            console.log(this.nombre); // undefined
        }, 1000);
    }
};

// PASO 3: Nosotros llamamos: persona.saludarDespues()
// Dentro de saludarDespues, "this" es persona
// Pero la función callback olvida quién es "persona"

    La clave: Cuando pasas una función a setTimeout, esa función "olvida" de dónde venía. Es como darle tu teléfono a alguien sin decirle tu nombre.

    Solución 1: Guardar this en una Variable

    let persona = {
    nombre: "Ana",
    saludarDespues: function() {
        // Guardar this en una variable
        let self = this; // o 'that', '_this', etc.
        
        setTimeout(function() {
            // ✅ Usar self en lugar de this
            console.log(`Hola, soy ${self.nombre}`);
        }, 1000);
    }
};

persona.saludarDespues(); // ✅ "Hola, soy Ana"

    Solución 2: Arrow Functions (Mejor)

    let persona = {
    nombre: "Ana",
    saludarDespues: function() {
        setTimeout(() => {
            // ✅ Arrow function mantiene el this del scope padre
            console.log(`Hola, soy ${this.nombre}`);
        }, 1000);
    }
};

persona.saludarDespues(); // ✅ "Hola, soy Ana"

    Solución 3: bind()

    let persona = {
    nombre: "Ana",
    saludarDespues: function() {
        setTimeout(function() {
            console.log(`Hola, soy ${this.nombre}`);
        }.bind(this), 1000); // ✅ bind() fuerza el this correcto
    }
};

persona.saludarDespues(); // ✅ "Hola, soy Ana"

    Ejemplo con Event Listeners:

    class Contador {
    constructor() {
        this.cuenta = 0;
        this.boton = document.getElementById('boton');
        
        // ❌ Problema: this se pierde
        this.boton.addEventListener('click', function() {
            this.cuenta++; // ❌ this es el botón, no Contador
            console.log(this.cuenta);
        });
        
        // ✅ Solución 1: Arrow function
        this.boton.addEventListener('click', () => {
            this.cuenta++; // ✅ this es Contador
            console.log(this.cuenta);
        });
        
        // ✅ Solución 2: bind()
        this.boton.addEventListener('click', function() {
            this.cuenta++;
            console.log(this.cuenta);
        }.bind(this));
    }
}

    Ejemplo con Métodos de Arrays:

    let persona = {
    nombre: "Ana",
    amigos: ["Carlos", "María", "Juan"],
    
    saludarAmigos: function() {
        // ❌ Problema
        this.amigos.forEach(function(amigo) {
            console.log(`${this.nombre} saluda a ${amigo}`);
            // this.nombre es undefined aquí
        });
        
        // ✅ Solución 1: Arrow function
        this.amigos.forEach(amigo => {
            console.log(`${this.nombre} saluda a ${amigo}`);

            console.log(`${this.nombre} saluda a ${amigo}`);
        }, this); // Pasar this como segundo argumento
    }
};

persona.saludarAmigos();

    Regla General:

    • Funciones normales: this depende de cómo se llama
    • Arrow functions: Heredan this del scope padre
    • En callbacks: Preferir arrow functions para mantener this

    Comparación:

    valor: 42,
    
    // Función normal - this puede cambiar

            console.log(this.valor); // undefined
        }, 1000);
    },

        console.log(this.valor); // 42
        
        setTimeout(() => {
            console.log(this.valor); // 42
        }, 1000);
    }
};

    🐛 Errores Comunes y Cómo Identificarlos

    Cuando this no funciona, verás estos síntomas:

    Síntoma 1: "undefined is not a function"

    let persona = {
    nombre: "Ana",
    saludar: function() {
        setTimeout(function() {
            this.nombre.toUpperCase(); // ❌ Error!
            // "this.nombre" es undefined
            // No puedes llamar .toUpperCase() en undefined
        }, 1000);
    }
};

    Causa: Perdiste el this, entonces this.nombre es undefined.

    Síntoma 2: Propiedades undefined

    let persona = {
    nombre: "Ana",
    saludar: function() {
        setTimeout(function() {
            console.log(this.nombre); // undefined
        }, 1000);
    }
};

persona.saludar(); // undefined (no "Ana")

    Causa: this no es lo que esperas (es window en lugar de persona).

    Cómo Debuggear - Usa console.log(this)

    let persona = {
    nombre: "Ana",
    saludar: function() {
        console.log("this en saludar:", this); // { nombre: "Ana", ... }
        
        setTimeout(function() {
            console.log("this en callback:", this); // Window o undefined
            // ¡Ahora ves el problema!
        }, 1000);
    }
};

    Truco: Siempre agrega console.log(this) cuando algo no funciona. Te mostrará exactamente qué es this en ese momento.

    ✅ Cuándo NO Preocuparte por "this"

    Hay casos donde this no importa y puedes relajarte:

    1. Funciones que no usan "this"

    function sumar(a, b) {
    return a + b; // No usa "this" → No hay problema
}

    3. Funciones auxiliares independientes

    // Estas funciones no necesitan un objeto:

function validarEmail(email) { ... }
function calcularPromedio(numeros) { ... }

    Arrow Functions en Callbacks

    ¿Por qué Arrow Functions para Callbacks?

    Las arrow functions son perfectas para callbacks porque:

    • Sintaxis más corta y clara
    • Mantienen el this del scope padre
    • Return implícito para expresiones simples

    Comparación: Función Normal vs Arrow:

    // Función normal
let numeros = [1, 2, 3, 4, 5];

let dobles1 = numeros.map(function(n) {
    return n * 2;
});

// Arrow function
let dobles2 = numeros.map(n => n * 2);

console.log(dobles1); // [2, 4, 6, 8, 10]
console.log(dobles2); // [2, 4, 6, 8, 10]

    setTimeout con Arrow Function:

    // ❌ Función normal
setTimeout(function() {
    console.log("Hola desde el futuro");
}, 1000);

// ✅ Arrow function (más corto)
setTimeout(() => {
    console.log("Hola desde el futuro");
}, 1000);

// ✅ Aún más corto (una línea)
setTimeout(() => console.log("Hola desde el futuro"), 1000);

    Event Listeners con Arrow Function:

    let boton = document.getElementById('miBoton');

// Función normal
boton.addEventListener('click', function(event) {
    console.log("Click en:", event.target);
});

// Arrow function
boton.addEventListener('click', (event) => {
    console.log("Click en:", event.target);
});

// Si solo hay un parámetro, puedes omitir los paréntesis
boton.addEventListener('click', event => {
    console.log("Click en:", event.target);
});

    Métodos de Arrays con Arrow Functions:

    let productos = [
    { nombre: 'Laptop', precio: 1000 },
    { nombre: 'Mouse', precio: 25 },
    { nombre: 'Teclado', precio: 75 }
];

// map
let nombres = productos.map(p => p.nombre);
console.log(nombres); // ['Laptop', 'Mouse', 'Teclado']

// filter
let caros = productos.filter(p => p.precio > 50);
console.log(caros);

// forEach
productos.forEach(p => {
    console.log(`${p.nombre}: $${p.precio}`);
});

// find
let laptop = productos.find(p => p.nombre === 'Laptop');
console.log(laptop);

    Mantener this con Arrow Functions:

    class ContadorClicks {
    constructor() {
        this.clicks = 0;
        this.boton = document.getElementById('boton');
        
        // ✅ Arrow function mantiene this
        this.boton.addEventListener('click', () => {
            this.clicks++;
            console.log(`Clicks: ${this.clicks}`);
        });

        // this.boton.addEventListener('click', function() {
        //     this.clicks++; // Error: this no es ContadorClicks
        // });
    }
}

    Return Implícito:

    let numeros = [1, 2, 3, 4, 5];

// Con return explícito

    return n * n;
});

// Con return implícito (más corto)
let cuadrados2 = numeros.map(n => n * n);

    Ejemplo Completo: Sistema de Tareas

    let tareas = [
    { id: 1, texto: 'Estudiar', completada: false },
    { id: 2, texto: 'Ejercicio', completada: true },
    { id: 3, texto: 'Leer', completada: false }
];

console.log("Número completadas:", numCompletadas);

    Promesas con Arrow Functions:

    // Tradicional
fetch('https://api.example.com/data')
    .then(function(response) {
        return response.json();
    })
    .then(function(data) {
        console.log(data);
    });

// Con arrow functions (mucho más limpio)
fetch('https://api.example.com/data')
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

    Cuándo usar Arrow Functions:

    • Callbacks de arrays: map, filter, forEach, etc.
    • Promesas: then(), catch()
    • Temporizadores: setTimeout, setInterval
    • Event listeners: Cuando necesitas acceder a this del scope padre
    • Funciones cortas: Cuando la lógica es simple

    Cuándo NO usar Arrow Functions:

    • Métodos de objetos: Si necesitas acceder a this del objeto
    • Constructores: No pueden usarse con new
    • prototype methods: Cuando defines métodos en prototipos
