// Crear botones
const botones = [
    ['C', 'CE', '/', '*'],
    ['7', '8', '9', '-'],
    ['4', '5', '6', '+'],
    ['1', '2', '3', '='],
    ['0', '.', '', '']
];

    const filaDiv = document.createElement('div');
    filaDiv.style.display = 'flex';
    filaDiv.style.gap = '5px';
    filaDiv.style.marginBottom = '5px';
    
    fila.forEach(texto => {

            filaDiv.appendChild(boton);
        } else {

function manejarClick(valor) {
    if (valor >= '0' && valor <= '9' || valor === '.') {
        if (!operacion) {
            numero1 += valor;
            display.value = numero1;
        } else {
            numero2 += valor;
            display.value = numero2;
        }
    } else if (['+', '-', '*', '/'].includes(valor)) {
        operacion = valor;
    } else if (valor === '=') {
        calcular();
    } else if (valor === 'C') {
        limpiar();
    } else if (valor === 'CE') {
        display.value = '';
    }
}

function calcular() {
    const n1 = parseFloat(numero1);
    const n2 = parseFloat(numero2);
    let resultado = 0;
    
    switch (operacion) {
        case '+': resultado = n1 + n2; break;
        case '-': resultado = n1 - n2; break;
        case '*': resultado = n1 * n2; break;
        case '/': 
            if (n2 === 0) {
                display.value = 'Error: Divisi贸n por cero';
                return;
            }
            resultado = n1 / n2; 
            break;
    }
    
    display.value = resultado;
    numero1 = resultado.toString();
    numero2 = '';
    operacion = '';
}

function limpiar() {
    numero1 = '';
    numero2 = '';
    operacion = '';
    display.value = '';
}

document.body.appendChild(calculadora);
