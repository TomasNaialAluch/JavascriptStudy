Selección de Elementos

Aprende los diferentes métodos para seleccionar elementos del DOM y trabajar con ellos en JavaScript.

🎯 Puntos Clave

Métodos de selección
Selectores CSS
Colecciones de elementos

Teoría

Métodos de Selección

getElementById() Selecciona por ID único
querySelector() Selecciona con CSS selector
querySelectorAll() Selecciona múltiples elementos
getElementsByClassName() Selecciona por clase

Selectores CSS

#id Por ID
.clase Por clase
elemento Por etiqueta
[atributo] Por atributo

Ejemplo Práctico

Selección Básica 
// Selección por ID
const titulo = document.getElementById('titulo');
console.log('Elemento por ID:', titulo);

// Selección por clase
const parrafos = document.getElementsByClassName('parrafo');
console.log('Elementos por clase:', parrafos);

// Selección por etiqueta
const divs = document.getElementsByTagName('div');
console.log('Elementos div:', divs);

// Selección con querySelector
const primerParrafo = document.querySelector('.parrafo');
console.log('Primer párrafo:', primerParrafo);

// Selección múltiple con querySelectorAll
const todosParrafos = document.querySelectorAll('.parrafo');
console.log('Todos los párrafos:', todosParrafos);

// Selección por atributo
const enlaces = document.querySelectorAll('a[href]');
console.log('Enlaces con href:', enlaces);
Selección Avanzada 
// Selectores complejos
const elementos = document.querySelectorAll('div.contenido > p:first-child');
console.log('Párrafos hijos directos:', elementos);

// Selección por atributos específicos
const inputs = document.querySelectorAll('input[type="text"]');
console.log('Inputs de texto:', inputs);

// Selección por pseudo-clases
const hoverElements = document.querySelectorAll('button:hover');
console.log('Botones en hover:', hoverElements);

// Navegación en el DOM
const contenedor = document.querySelector('.contenedor');
const hijos = contenedor.children;
const padre = contenedor.parentElement;
const hermanos = contenedor.nextElementSibling;

console.log('Hijos del contenedor:', hijos);
console.log('Elemento padre:', padre);
console.log('Hermano siguiente:', hermanos);

// Verificar si existe un elemento
const existe = document.getElementById('no-existe');
if (existe) {
    console.log('El elemento existe');
} else {
    console.log('El elemento no existe');
}

Eventos Básicos

const boton = document.createElement('button');
boton.textContent = 'Haz clic aquí';

    console.log('¡Botón clickeado!');

    console.log('Texto ingresado:', event.target.value);

    if (event.key === 'Enter') {
        console.log('Enter presionado!');
    }
});

// Evento de focus
input.addEventListener('focus', function() {
    this.style.border = '2px solid blue';
});

// Evento de blur
input.addEventListener('blur', function() {
    this.style.border = '1px solid gray';
});

console.log('Eventos configurados. Prueba el botón y el input.');
Event Delegation 
// Crear lista dinámica
const lista = document.createElement('ul');
lista.id = 'miLista';
document.body.appendChild(lista);

// Agregar algunos elementos
for (let i = 1; i <= 3; i++) {
    const li = document.createElement('li');
    li.textContent = `Elemento ${i}`;
    li.className = 'item-lista';
    li.style.cursor = 'pointer';
    li.style.padding = '5px';
    li.style.margin = '2px';
    li.style.border = '1px solid #ccc';
    lista.appendChild(li);
}

// Event delegation - un solo listener para todos los elementos
lista.addEventListener('click', function(event) {
    // Verificar si el elemento clickeado es un li
    if (event.target.classList.contains('item-lista')) {
        console.log('Elemento clickeado:', event.target.textContent);
        event.target.style.backgroundColor = 'yellow';
    }
});

// Evento de mouseover
lista.addEventListener('mouseover', function(event) {
    if (event.target.classList.contains('item-lista')) {
        event.target.style.backgroundColor = 'lightgreen';
    }
});

// Evento de mouseout
lista.addEventListener('mouseout', function(event) {
    if (event.target.classList.contains('item-lista')) {
        event.target.style.backgroundColor = '';
    }
});

console.log('Event delegation configurado. Haz clic en los elementos de la lista.');

Ejemplo Práctico

Formulario de Contacto Interactivo

// Formulario de contacto con validación
const form = document.getElementById('contactForm');
const inputs = form.querySelectorAll('input, textarea');

// Validación en tiempo real
inputs.forEach(input => {
    input.addEventListener('blur', function() {
        validarCampo(this);
    });
    
    input.addEventListener('input', function() {
        if (this.classList.contains('error')) {
            validarCampo(this);
        }
    });
});

function validarCampo(campo) {
    const valor = campo.value.trim();
    let esValido = true;
    let mensaje = '';

    if (campo.hasAttribute('required') && !valor) {
        esValido = false;
        mensaje = 'Este campo es obligatorio';
    } else if (campo.type === 'email' && valor) {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
        if (!emailRegex.test(valor)) {
            esValido = false;
            mensaje = 'Ingresa un email válido';
        }
    }

    if (esValido) {
        campo.classList.remove('error');
        campo.setCustomValidity('');
        ocultarError(campo);
    } else {
        campo.classList.add('error');
        campo.setCustomValidity(mensaje);
        mostrarError(campo, mensaje);
    }
}

// Envío del formulario
form.addEventListener('submit', function(e) {
    e.preventDefault();
    
    const formData = new FormData(this);
    const datos = Object.fromEntries(formData);
    
    console.log('Datos del formulario:', datos);
    alert('¡Formulario enviado correctamente!');
    form.reset();
    limpiarErrores();
});

function mostrarError(campo, mensaje) {
    let errorDiv = campo.parentNode.querySelector('.error-message');
    if (!errorDiv) {
        errorDiv = document.createElement('div');
        errorDiv.className = 'error-message';
        campo.parentNode.appendChild(errorDiv);
    }
    errorDiv.textContent = mensaje;
}

function ocultarError(campo) {
    const errorDiv = campo.parentNode.querySelector('.error-message');
    if (errorDiv) {
        errorDiv.remove();
    }
}

function limpiarErrores() {
    inputs.forEach(input => {
        input.classList.remove('error');
        input.setCustomValidity('');
        ocultarError(input);
    });
}

// Crear elemento animado
const elemento = document.createElement('div');
elemento.className = 'elemento-animado';
elemento.textContent = 'Hover me!';
elemento.style.cssText = `
    width: 100px;
    height: 100px;
    background: linear-gradient(45deg, #ff6b6b, #4ecdc4);
    border-radius: 50%;
    display: flex;
    align-items: center;
    justify-content: center;
    color: white;
    font-weight: bold;
    cursor: pointer;
    transition: all 0.3s ease;
    margin: 20px auto;
`;

// Agregar al DOM
document.body.appendChild(elemento);

// Eventos de hover
elemento.addEventListener('mouseenter', function() {
    this.style.transform = 'scale(1.2) rotate(180deg)';
    this.style.boxShadow = '0 10px 20px rgba(0,0,0,0.3)';
});

elemento.addEventListener('mouseleave', function() {
    this.style.transform = 'scale(1) rotate(0deg)';
    this.style.boxShadow = 'none';
});

// Animación con requestAnimationFrame
let posicion = 0;
let animando = false;

function animarElemento() {
    if (!animando) return;
    
    posicion += 2;
    elemento.style.transform = `translateX(${posicion}px) scale(1.1)`;
    
    if (posicion < 300) {
        requestAnimationFrame(animarElemento);
    } else {
        posicion = 0;
        animando = false;
        elemento.style.transform = 'translateX(0) scale(1)';
    }
}

// Botones de control
const botonIniciar = document.createElement('button');
botonIniciar.textContent = 'Iniciar Animación';
botonIniciar.style.cssText = 'margin: 10px; padding: 10px 20px; background: #007bff; color: white; border: none; border-radius: 5px; cursor: pointer;';
botonIniciar.addEventListener('click', function() {
    animando = true;
    animarElemento();
});

const botonPulso = document.createElement('button');
botonPulso.textContent = 'Efecto Pulso';
botonPulso.style.cssText = 'margin: 10px; padding: 10px 20px; background: #28a745; color: white; border: none; border-radius: 5px; cursor: pointer;';
botonPulso.addEventListener('click', function() {
    elemento.classList.toggle('pulso');
});

// Agregar estilos CSS
const style = document.createElement('style');
style.textContent = `
    @keyframes pulso {
        0%, 100% { transform: scale(1); }
        50% { transform: scale(1.1); }
    }
    .pulso {
        animation: pulso 1s ease-in-out infinite;
    }
`;
document.head.appendChild(style);

// Agregar botones al DOM
document.body.appendChild(botonIniciar);
document.body.appendChild(botonPulso);

// Gestor de tareas con localStorage
let tareas = JSON.parse(localStorage.getItem('tareas')) || [];

function agregarTarea(texto) {
    const tarea = {
        id: Date.now(),
        texto: texto,
        completada: false,
        fecha: new Date().toISOString()
    };
    
    tareas.push(tarea);
    guardarTareas();
    cargarTareas();
}

function completarTarea(id) {
    const tarea = tareas.find(t => t.id === id);
    if (tarea) {
        tarea.completada = !tarea.completada;
        guardarTareas();
        cargarTareas();
    }
}

function eliminarTarea(id) {
    tareas = tareas.filter(t => t.id !== id);
    guardarTareas();
    cargarTareas();
}

function guardarTareas() {
    localStorage.setItem('tareas', JSON.stringify(tareas));
}

function cargarTareas() {
    const contenedor = document.getElementById('lista-tareas');

    contenedor.innerHTML = tareas.map(tarea => `
        

            ${tarea.texto}
            

                
                
            

        

    `).join('');
}

const elemento = document.getElementById('miElemento');
let animando = false;

// Función de animación suave
function animarElemento() {
    if (animando) return;
    animando = true;
    
    let posicion = 0;
    const velocidad = 2;
    
    function mover() {
        posicion += velocidad;
        elemento.style.transform = `translateX(${posicion}px)`;
        
        if (posicion < 200) {
            requestAnimationFrame(mover);
        } else {
            // Volver a la posición original
            elemento.style.transition = 'transform 0.5s ease';
            elemento.style.transform = 'translateX(0)';
            setTimeout(() => {
                elemento.style.transition = '';
                animando = false;
            }, 500);
        }
    }
    
    mover();
}

    elemento.classList.add('hover-effect');

Ejemplo Práctico

Lista de Tareas Persistente

// HTML: <input id="tareaInput" placeholder="Nueva tarea">
//       <button onclick="agregarTarea()">Agregar</button>
//       <ul id="listaTareas"></ul>

let tareas = JSON.parse(localStorage.getItem('tareas')) || [];

// Cargar tareas al iniciar
cargarTareas();

function agregarTarea() {
    const input = document.getElementById('tareaInput');
    const texto = input.value.trim();
    
    if (texto) {
        const tarea = {
            id: Date.now(),
            texto: texto,
            completada: false
        };
        
        tareas.push(tarea);
        guardarTareas();
        cargarTareas();
        input.value = '';
    }
}

function cargarTareas() {
    const lista = document.getElementById('listaTareas');
    lista.innerHTML = '';
    
    tareas.forEach(tarea => {
        const li = document.createElement('li');
        li.innerHTML = `
            <span class="${tarea.completada ? 'completada' : ''}">
                ${tarea.texto}
            </span>
            <button onclick="toggleTarea(${tarea.id})">
                ${tarea.completada ? 'Deshacer' : 'Completar'}
            </button>
            <button onclick="eliminarTarea(${tarea.id})">Eliminar</button>
        `;
        lista.appendChild(li);
    });
}

function toggleTarea(id) {
    const tarea = tareas.find(t => t.id === id);
    if (tarea) {
        tarea.completada = !tarea.completada;
        guardarTareas();
        cargarTareas();
    }
}

function eliminarTarea(id) {
    tareas = tareas.filter(t => t.id !== id);
    guardarTareas();
    cargarTareas();
}

function guardarTareas() {
    localStorage.setItem('tareas', JSON.stringify(tareas));
}

console.log(input.validity.valueMissing); // true si está vacío
console.log(input.validity.typeMismatch); // true si el tipo no coincide

Propiedades principales:

  • valid - true si el elemento es válido
  • valueMissing - true si el campo es requerido pero está vacío
  • typeMismatch - true si el valor no coincide con el tipo
  • patternMismatch - true si no coincide con el patrón

Evento Submit

×

¿Qué es el evento submit?

Se dispara cuando un formulario es enviado, ya sea por clic en botón submit o presionar Enter.

Sintaxis:

formulario.addEventListener('submit', function(e) {
    // código aquí
});

Ejemplo:

const form = document.getElementById('miFormulario');

form.addEventListener('submit', function(e) {
    e.preventDefault(); // Prevenir envío por defecto
    
    const formData = new FormData(this);
    const datos = Object.fromEntries(formData);
    
    console.log('Datos del formulario:', datos);
    
    // Validar datos
    if (validarFormulario(datos)) {
        enviarDatos(datos);
    }
});

Características:

  • Se dispara antes de la validación HTML5
  • Se puede prevenir con preventDefault()
  • Útil para validación personalizada
  • Permite procesar datos antes del envío

preventDefault()

×

¿Qué es preventDefault()?

Previene el comportamiento por defecto de un evento, como el envío automático de un formulario.

Sintaxis:

evento.preventDefault();

Ejemplo:

form.addEventListener('submit', function(e) {
    e.preventDefault(); // Evita que se envíe el formulario
    
    // Validación personalizada
    if (validarDatos()) {
        // Enviar manualmente
        enviarFormulario();
    } else {
        mostrarErrores();
    }
});

Casos de uso:

  • Validación personalizada antes del envío
  • Procesamiento de datos con JavaScript
  • Enviar datos vía AJAX en lugar de recargar la página
  • Mostrar confirmaciones antes de enviar

setCustomValidity()

×

¿Qué es setCustomValidity()?

Establece un mensaje de validación personalizado para un elemento de formulario.

Sintaxis:

elemento.setCustomValidity('mensaje');

Ejemplo:

const email = document.getElementById('email');

email.addEventListener('blur', function() {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    
    if (!emailRegex.test(this.value)) {
        this.setCustomValidity('Por favor, ingresa un email válido');
    } else {
        this.setCustomValidity(''); // Limpiar mensaje
    }

Características:

  • Mensaje vacío ('') marca el campo como válido
  • Cualquier texto marca el campo como inválido
  • Se muestra en el tooltip de validación del navegador
  • Útil para validaciones complejas

FormData

×

¿Qué es FormData?

Es un objeto que facilita la recolección y envío de datos de formularios, incluyendo archivos.

Sintaxis:

const formData = new FormData(formulario);

const form = document.getElementById('miFormulario');
const formData = new FormData(form);

Métodos principales:

  • get(nombre) - Obtiene el valor de un campo
  • set(nombre, valor) - Establece un valor
  • append(nombre, valor) - Añade un valor
  • has(nombre) - Verifica si existe un campo

Restaura un formulario a su estado inicial, limpiando todos los campos y validaciones.

form.addEventListener('submit', function(e) {

Propiedad Transition

¿Qué es transition?

Permite crear transiciones suaves entre diferentes estados de un elemento CSS.

Sintaxis:

transition: propiedad duración función tiempo-inicio;

Propiedad Transform

×

¿Qué es transform?

Permite modificar la forma, posición y orientación de un elemento sin afectar el flujo del documento.

Sintaxis:

transform: función(valor);

Funciones principales:

  • scale(x, y) - Escalar elemento

@keyframes

×

¿Qué es @keyframes?

Define los pasos de una animación CSS, especificando qué propiedades cambiar y cuándo.

@keyframes nombre-animacion {

  • animation-direction - Dirección

setInterval()

×

¿Qué es setInterval()?

Ejecuta una función repetidamente con un intervalo de tiempo fijo entre cada ejecución.

Sintaxis:

setInterval(función, milisegundos);

Ejemplo:

let posicion = 0;
const elemento = document.getElementById('miElemento');

const animacion = setInterval(() => {
    posicion += 5;
    elemento.style.left = posicion + 'px';
    
    if (posicion > 500) {
        clearInterval(animacion); // Detener animación
    }
}, 50); // Cada 50ms

Características:

  • Devuelve un ID para poder detenerlo
  • Se detiene con clearInterval()
  • Útil para animaciones simples
  • Puede causar problemas de rendimiento

requestAnimationFrame()

×

¿Qué es requestAnimationFrame()?

Método optimizado para crear animaciones suaves que se sincroniza con el refresco de la pantalla.

Sintaxis:

requestAnimationFrame(función);

Ejemplo:

let posicion = 0;
const elemento = document.getElementById('miElemento');

function animar() {
    posicion += 2;
    elemento.style.transform = `translateX(${posicion}px)`;
    
    if (posicion < 500) {
        requestAnimationFrame(animar); // Continuar animación
    }
}

requestAnimationFrame(animar); // Iniciar animación

Ventajas sobre setInterval:

  • Mejor rendimiento y fluidez
  • Se pausa automáticamente en pestañas inactivas
  • Sincronizado con el refresco de pantalla
  • Recomendado para animaciones complejas

setItem()

×

¿Qué es setItem()?

Guarda un par clave-valor en el almacenamiento local del navegador.

Sintaxis:

localStorage.setItem('clave', 'valor');

Ejemplo:

// Guardar datos simples
localStorage.setItem('nombre', 'Juan');
localStorage.setItem('edad', '25');

// Guardar objetos (convertir a JSON)
const usuario = { nombre: 'Juan', edad: 25, email: 'juan@email.com' };

Características:

  • Los datos persisten entre sesiones
  • Solo almacena strings
  • Límite de ~5-10MB por dominio
  • Se puede sobrescribir con la misma clave

getItem()

×

¿Qué es getItem()?

Recupera un valor del almacenamiento local usando su clave.

Sintaxis:

localStorage.getItem('clave');

Ejemplo:

// Recuperar datos simples
const nombre = localStorage.getItem('nombre');
const edad = localStorage.getItem('edad');

// Recuperar objetos (convertir desde JSON)
const usuarioStr = localStorage.getItem('usuario');
if (usuarioStr) {
    const usuario = JSON.parse(usuarioStr);
    console.log(usuario.nombre); // Juan
}

Características:

  • Devuelve null si la clave no existe
  • Siempre devuelve un string
  • Necesita conversión para objetos
  • Verificar existencia antes de usar

removeItem()

×

¿Qué es removeItem()?

Elimina un elemento específico del almacenamiento local usando su clave.

Sintaxis:

localStorage.removeItem('clave');

Ejemplo:

// Eliminar un elemento específico
localStorage.removeItem('usuario');

// Verificar si se eliminó
const usuario = localStorage.getItem('usuario');
console.log(usuario); // null

// Eliminar múltiples elementos
const clavesAEliminar = ['temporal', 'cache', 'logs'];
clavesAEliminar.forEach(clave => {
    localStorage.removeItem(clave);
});

Características:

  • No causa error si la clave no existe
  • Elimina solo el elemento especificado
  • No afecta otros datos almacenados
  • Útil para limpiar datos específicos

clear()

×

¿Qué es clear()?

Elimina todos los elementos del almacenamiento local del dominio actual.

Sintaxis:

localStorage.clear();

Ejemplo:

// Guardar algunos datos
localStorage.setItem('usuario', 'Juan');
localStorage.setItem('config', 'dark');
localStorage.setItem('preferencias', 'es');

// Verificar que existen
console.log(localStorage.length); // 3

// Limpiar todo
localStorage.clear();
console.log(localStorage.length); // 0

Casos de uso:

  • Logout de usuario
  • Reset de configuración
  • Limpieza de datos temporales
  • Pruebas y debugging

JSON.stringify()

×

¿Qué es JSON.stringify()?

Convierte un objeto JavaScript a una cadena JSON para poder guardarlo en localStorage.

Sintaxis:

JSON.stringify(objeto);

sessionStorage

×

¿Qué es sessionStorage?

Similar a localStorage pero los datos se eliminan cuando se cierra la pestaña del navegador.

Sintaxis:

sessionStorage.setItem('clave', 'valor');
sessionStorage.getItem('clave');
sessionStorage.removeItem('clave');

Ejemplo:

// Guardar datos de sesión
sessionStorage.setItem('carrito', JSON.stringify(['item1', 'item2']));
sessionStorage.setItem('ultimaPagina', 'productos');

// Los datos persisten mientras la pestaña esté abierta
const carrito = JSON.parse(sessionStorage.getItem('carrito'));
console.log(carrito); // ['item1', 'item2']

// Se eliminan al cerrar la pestaña

Diferencias con localStorage:

  • localStorage: Persiste hasta ser eliminado
  • sessionStorage: Se elimina al cerrar pestaña
  • Misma API (setItem, getItem, etc.)
  • Útil para datos temporales
