Debugging Avanzado 
// Función con debugging
function procesarDatos(datos) {
    console.log('Iniciando procesamiento...');
    console.log('Datos recibidos:', datos);
    
    // Validar datos
    if (!Array.isArray(datos)) {
        console.error('Error: Los datos deben ser un array');
        return;
    }
    
    console.log('Datos válidos, procesando...');
    
    // Procesar cada elemento
    const resultado = datos.map((item, index) => {
        console.log(`Procesando item ${index}:`, item);
        
        // Simular procesamiento
        const procesado = {
            id: index + 1,
            valor: item * 2,
            timestamp: Date.now()
        };
        
        console.log(`Item ${index} procesado:`, procesado);
        return procesado;
    });
    
    console.log('Procesamiento completado');
    console.table(resultado);
    
    return resultado;
}

// Función con breakpoints condicionales
function buscarUsuario(usuarios, id) {
    console.log('Buscando usuario con ID:', id);
    
    for (let i = 0; i < usuarios.length; i++) {
        const usuario = usuarios[i];
        console.log(`Verificando usuario ${i}:`, usuario);
        
        // Breakpoint condicional: solo si el ID coincide
        if (usuario.id === id) {
            console.log('¡Usuario encontrado!', usuario);
            return usuario;
        }
    }
    
    console.warn('Usuario no encontrado');
    return null;
}

// Función para testing de performance
function testPerformance() {
    console.log('Iniciando test de performance...');
    
    // Test 1: Array methods
    console.time('Array methods');
    const numeros = Array.from({length: 100000}, (_, i) => i);
    const dobles = numeros.map(n => n * 2);
    const pares = dobles.filter(n => n % 2 === 0);
    console.timeEnd('Array methods');
    
    // Test 2: Loop tradicional
    console.time('Loop tradicional');
    const resultado = [];
    for (let i = 0; i < 100000; i++) {
        const doble = i * 2;
        if (doble % 2 === 0) {
            resultado.push(doble);
        }
    }
    console.timeEnd('Loop tradicional');
    
    console.log('Test de performance completado');
}

// Función para monitorear memoria
function monitorearMemoria() {
    if (performance.memory) {
        console.log('Memoria utilizada:', performance.memory.usedJSHeapSize / 1024 / 1024, 'MB');
        console.log('Memoria total:', performance.memory.totalJSHeapSize / 1024 / 1024, 'MB');
        console.log('Límite de memoria:', performance.memory.jsHeapSizeLimit / 1024 / 1024, 'MB');
    } else {
        console.log('Información de memoria no disponible');
    }
}

Git y GitHub

Control de versiones esencial para cualquier desarrollador. Aprende Git desde lo básico hasta flujos de trabajo avanzados.

🎯 Puntos Clave

Comandos básicos
Branches y merge
GitHub workflow

Teoría

Comandos Básicos

git init Inicializar repositorio
git add Agregar archivos al staging
git commit Crear commit
git push Enviar cambios al remoto

Branches y Merge

git branch Listar/crear branches
git checkout Cambiar de branch
git merge Fusionar branches
git rebase Reescribir historial

Ejemplo Práctico

Flujo Básico de Git 
# Inicializar un nuevo repositorio
git init mi-proyecto
cd mi-proyecto

# Configurar usuario (solo la primera vez)
git config --global user.name "Tu Nombre"
git config --global user.email "tu@email.com"

# Verificar estado del repositorio
git status

# Crear archivo README
echo "# Mi Proyecto" > README.md
echo "Este es mi primer proyecto con Git" >> README.md

# Agregar archivos al staging area
git add README.md

# Ver qué archivos están en staging
git status

# Crear el primer commit
git commit -m "feat: agregar README inicial"

# Ver el historial de commits
git log --oneline

# Crear más archivos
echo "console.log('Hola Mundo');" > app.js
echo "body { background: #f0f0f0; }" > styles.css

# Agregar todos los archivos
git add .

# Commit con múltiples archivos
git commit -m "feat: agregar archivos de aplicación"

# Ver el historial completo
git log --oneline --graph

# Crear un archivo .gitignore
echo "node_modules/" > .gitignore
echo "*.log" >> .gitignore
echo ".env" >> .gitignore

# Agregar .gitignore
git add .gitignore
git commit -m "chore: agregar .gitignore"

# Ver el estado actual
git status
git log --oneline
Flujo Avanzado con Branches 
# Crear y cambiar a nueva rama
git checkout -b feature/nueva-funcionalidad

# Hacer cambios en la nueva rama
echo "function nuevaFuncion() {" >> app.js
echo "  console.log('Nueva funcionalidad');" >> app.js
echo "}" >> app.js

# Commit en la rama feature
git add app.js
git commit -m "feat: agregar nueva funcionalidad"

# Crear otra rama para bugfix
git checkout main
git checkout -b bugfix/corregir-error

# Corregir un error
echo "// Bug corregido" >> app.js
git add app.js
git commit -m "fix: corregir error en app.js"

# Volver a main y fusionar feature
git checkout main
git merge feature/nueva-funcionalidad

# Fusionar bugfix
git merge bugfix/corregir-error

# Ver el historial con branches
git log --oneline --graph --all

# Eliminar ramas ya fusionadas
git branch -d feature/nueva-funcionalidad
git branch -d bugfix/corregir-error

# Configurar repositorio remoto (GitHub)
git remote add origin https://github.com/usuario/mi-proyecto.git

# Enviar cambios al remoto
git push -u origin main

# Crear nueva rama y enviarla
git checkout -b feature/otra-funcionalidad
echo "// Otra funcionalidad" >> app.js
git add app.js
git commit -m "feat: otra funcionalidad"
git push -u origin feature/otra-funcionalidad

# Ver ramas remotas
git branch -r

# Actualizar rama local con cambios remotos
git checkout main
git pull origin main

# Ver diferencias entre ramas
git diff main feature/otra-funcionalidad

# Ver estadísticas del repositorio
git log --stat --oneline

Flujo de Trabajo Git Completo

Intermedio

Implementa un flujo de trabajo Git completo con GitHub.

Requisitos:

  • Configurar repositorio local y remoto
  • Crear branches para features
  • Implementar pull requests
  • Configurar GitHub Actions
  • Manejar conflictos de merge
# 1. Configurar repositorio
git init
git remote add origin https://github.com/usuario/proyecto.git

# 2. Crear rama principal
git checkout -b main
echo "# Proyecto" > README.md
git add README.md
git commit -m "feat: inicializar proyecto"
git push -u origin main

# 3. Crear feature branch
git checkout -b feature/nueva-funcionalidad
echo "function nuevaFuncion() { return 'Hola'; }" > src/funciones.js
git add src/funciones.js
git commit -m "feat: agregar nueva función"
git push -u origin feature/nueva-funcionalidad

# 4. Crear pull request (desde GitHub)
# - Ir a GitHub
# - Crear Pull Request
# - Revisar cambios
# - Merge a main

# 5. Actualizar local después del merge
git checkout main
git pull origin main
git branch -d feature/nueva-funcionalidad

# 6. Crear hotfix branch
git checkout -b hotfix/corregir-bug
echo "// Bug corregido" >> src/funciones.js
git add src/funciones.js
git commit -m "fix: corregir bug crítico"
git push -u origin hotfix/corregir-bug

# 7. Merge hotfix
git checkout main
git merge hotfix/corregir-bug
git push origin main
git branch -d hotfix/corregir-bug

# 8. Configurar GitHub Actions (.github/workflows/ci.yml)
name: CI
on: [push, pull_request]
jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - uses: actions/setup-node@v2
        with:
          node-version: '18'
      - run: npm install
      - run: npm test
      - run: npm run lint
